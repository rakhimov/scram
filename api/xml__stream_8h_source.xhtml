46 /// Manages XML element indentation upon output.
47 ///
48 /// @pre Indenter instance is not shared between XML Streams.
49 ///
50 /// @note The indentation is bounded by kMaxIndent.
51 /// If indentation exceeds kMaxIndent,
52 /// only kMaxIndent indentation will be produced w/o error.
53 class Indenter {
54  public:
55  /// RAII class to manage the current indentation string.
56  class Indentation {
57  public:
58  /// @param[in] num_chars The number of indentation characters.
59  /// @param[in,out] indent The provider of the indentation buffer.
60  ///
61  /// @pre There's only a single instance at a time.
62  /// @pre The indent buffer is filled w/ kIndentChar up to kMaxIndent.
63  ///
64  /// @post If the number of chars exceeds kMaxIndent,
65  /// the indentation is kMaxIndent.
66  Indentation(int num_chars, Indenter* indent)
67  : pos_(num_chars > kMaxIndent ? kMaxIndent : num_chars),
68  indent_(indent) {
69  indent_->spaces[pos_] = '\0'; // Indent string up to the position.
70  }
71 
72  /// Restores the buffer.
73  ~Indentation() { indent_->spaces[pos_] = kIndentChar; }
74 
75  /// @returns The indentation string.
76  operator const char*() { return indent_->spaces; }
77 
78  private:
79  int pos_; ///< The position of the null char in the string.
80  Indenter* indent_; ///< The string buffer provider.
81  };
82 
83  /// Initializes the buffer with enough space characters.
84  ///
85  /// @param[in] indent Enable/disable indentation.
86  explicit Indenter(bool indent = true) : indent_(indent) {
87  std::fill_n(spaces, kMaxIndent, kIndentChar);
88  }
89 
90  /// @param[in] num_chars The number of indentation characters.
91  ///
92  /// @returns Indentation to produce the representative string.
93  Indentation operator()(int num_chars) {
94  return indent_ ? Indentation(num_chars, this) : Indentation(0, this);
95  }
96 
97  private:
98  bool indent_; ///< Option to enable/disable indentation.
99  char spaces[kMaxIndent + 1]; ///< The indentation and terminator.
100 };

155 /// Writer of data formed as an XML element to a stream.
156 /// This class relies on the RAII to put the closing tags.
157 /// It is designed for stack-based use
158 /// so that its destructor gets called at scope exit.
159 /// One element at a time must be operated.
160 /// The output will be malformed
161 /// if two child elements at the same scope are being streamed.
162 /// To prevent this from happening,
163 /// the parent element is put into an inactive state
164 /// while its child element is alive.
165 ///
166 /// @pre All strings are UTF-8 encoded.
167 ///
168 /// @note The stream is not designed to mix XML text and elements
169 /// due to the absence of the use case or need.
170 ///
171 /// @warning The names of elements and attributes
172 /// are NOT validated to be proper XML NCName.
173 /// It is up to the caller
174 /// to sanitize the names (<, >, &, ", ', :, etc.).
175 ///
176 /// @warning The API works with C strings,
177 /// but this class does not manage the string lifetime.
178 /// It doesn't own any strings.
179 /// The provider of the strings must make sure
180 /// the lifetime of the string is long enough for streaming.
181 /// It is the most common case that strings are literals,
182 /// so there's no need to worry about dynamic lifetime.
183 class StreamElement {
184  public:
185  /// Constructs a root streamer for the XML element data
186  /// ready to accept attributes, elements, text.
187  ///
188  /// @param[in] name Non-empty string name for the element.
189  /// @param[in] indenter The indentation provider.
190  /// @param[in,out] out The destination stream.
191  ///
192  /// @throws StreamError Invalid setup for the element.
193  StreamElement(const char* name, detail::Indenter* indenter,
194  detail::FileStream* out)
195  : StreamElement(name, 0, nullptr, indenter, out) {}
196 
197  /// Puts the closing tag.
198  ///
199  /// @pre No child element is alive.
200  ///
201  /// @warning The output will be malformed
202  /// if the child outlives the parent.
203  /// It can happen if the destructor is called explicitly,
204  /// or if the objects are allocated on the heap
205  /// with different lifetimes.
206  ~StreamElement() noexcept {
207  assert(active_ && "The child element may still be alive.");
208  assert(!(parent_ && parent_->active_) && "The parent must be inactive.");
209  if (parent_)
210  parent_->active_ = true;
211  if (accept_attributes_) {
212  out_ << "/>\n";
213  } else if (accept_elements_) {
214  out_ << indenter_(kIndent_);
215  closing_tag:
216  out_ << "</" << kName_ << ">\n";
217  } else {
218  assert(accept_text_ && "The element is in unspecified state.");
219  goto closing_tag;
220  }
221  }
222 
223  /// Sets the attributes for the element.
224  ///
225  /// @tparam T Streamable type supporting operator<<.
226  ///
227  /// @param[in] name Non-empty name for the attribute.
228  /// @param[in] value The value of the attribute.
229  ///
230  /// @returns The reference to this element.
231  ///
232  /// @post &, <, " chars are escaped in the value text.
233  ///
234  /// @throws StreamError Invalid setup for the attribute.
235  template <typename T>
236  StreamElement& SetAttribute(const char* name, T&& value) {
237  if (!active_)
238  throw StreamError("The element is inactive.");
239  if (!accept_attributes_)
240  throw StreamError("Too late for attributes.");
241  if (*name == '\0')
242  throw StreamError("Attribute name can't be empty.");
243 
244  out_ << " " << name << "=\"";
245  PutValue(std::forward<T>(value));
246  out_ << "\"";
247  return *this;
248  }
249 
250  /// Adds text to the element.
251  ///
252  /// @tparam T Streamable type supporting operator<<.
253  ///
254  /// @param[in] text Non-empty text.
255  ///
256  /// @returns The reference to this element.
257  ///
258  /// @post No more elements or attributes can be added.
259  /// @post More text can be added.
260  /// @post &, <, " chars are escaped in the text.
261  ///
262  /// @throws StreamError Invalid setup or state for text addition.
263  template <typename T>
264  StreamElement& AddText(T&& text) {
265  if (!active_)
266  throw StreamError("The element is inactive.");
267  if (!accept_text_)
268  throw StreamError("Too late to put text.");
269 
270  if (accept_elements_)
271  accept_elements_ = false;
272  if (accept_attributes_) {
273  accept_attributes_ = false;
274  out_ << ">";
275  }
276  PutValue(std::forward<T>(text));
277  return *this;
278  }
279 
280  /// Adds a child element to the element.
281  ///
282  /// @param[in] name Non-empty name for the child element.
283  ///
284  /// @returns A streamer for child element.
285  ///
286  /// @pre The child element will be destroyed before the parent.
287  ///
288  /// @post The parent element is inactive
289  /// while the child element is alive.
290  ///
291  /// @throws StreamError Invalid setup or state for element addition.
292  StreamElement AddChild(const char* name) {
293  if (!active_)
294  throw StreamError("The element is inactive.");
295  if (!accept_elements_)
296  throw StreamError("Too late to add elements.");
297  if (*name == '\0')
298  throw StreamError("Element name can't be empty.");
299 
300  if (accept_text_)
301  accept_text_ = false;
302  if (accept_attributes_) {
303  accept_attributes_ = false;
304  out_ << ">\n";
305  }
306  return StreamElement(name, kIndent_ + kIndentIncrement, this, &indenter_,
307  &out_);
308  }
309 
310  private:
311  static const int kIndentIncrement = 2; ///< The number of chars per indent.
312 
313  /// Private constructor for a streamer
314  /// to pass parent-child information.
315  ///
316  /// @param[in] name Non-empty string name for the element.
317  /// @param[in] indent The number of spaces to indent the tags.
318  /// @param[in,out] parent The parent element.
319  /// Null pointer for root streamers.
320  /// @param[in] indenter The indentation provider.
321  /// @param[in,out] out The destination stream.
322  ///
323  /// @throws StreamError Invalid setup for the element.
324  StreamElement(const char* name, int indent, StreamElement* parent,
325  detail::Indenter* indenter, detail::FileStream* out)
326  : kName_(name),
327  kIndent_(indent),
328  accept_attributes_(true),
329  accept_elements_(true),
330  accept_text_(true),
331  active_(true),
332  parent_(parent),
333  indenter_(*indenter),
334  out_(*out) {
335  if (*kName_ == '\0')
336  throw StreamError("The element name can't be empty.");
337 
338  if (parent_) {
339  if (!parent_->active_)
340  throw StreamError("The parent is inactive.");
341  parent_->active_ = false;
342  }
343  assert(kIndent_ >= 0 && "Negative XML indentation.");
344  out_ << indenter_(kIndent_) << "<" << kName_;
345  }

347  /// Puts the value as text escaping the required XML special characters.
348  /// @{
349  void PutValue(int value) { out_ << value; }
350  void PutValue(double value) { out_ << value; }
351  void PutValue(std::size_t value) { out_ << value; }
352  void PutValue(bool value) { out_ << (value ? "true" : "false"); }
353  void PutValue(const std::string& value) { PutValue(value.c_str()); }
354  void PutValue(const char* value) {
355  bool has_special_chars = [value]() mutable { // 2x faster than strpbrk.
356  for (;; ++value) {
357  switch (*value) {
358  case '\0':
359  return false;
360  case '&':
361  case '<':
362  case '"':
363  return true;
364  }
365  }
366  }();
367 
368  if (!has_special_chars) { // The most common case.
369  out_ << value;
370  } else {
371  for (;; ++value) {
372  switch (*value) {
373  case '\0':
374  return;
375  case '&':
376  out_ << "&amp;";
377  break;
378  case '<':
379  out_ << "&lt;";
380  break;
381  case '"':
382  out_ << "&quot;";
383  break;
384  default:
385  out_ << *value;
386  }
387  }
388  }
389  }
390  /// @}

392  const char* kName_; ///< The name of the element.
393  const int kIndent_; ///< Indentation for tags.
394  bool accept_attributes_; ///< Flag for preventing late attributes.
395  bool accept_elements_; ///< Flag for preventing late elements.
396  bool accept_text_; ///< Flag for preventing late text additions.
397  bool active_; ///< Active in streaming.
398  StreamElement* parent_; ///< Parent element.
399  detail::Indenter& indenter_; ///< The indentation string producer.
400  detail::FileStream& out_; ///< The output destination.

403 /// XML Stream document.
404 ///
405 /// @pre Only this stream and its elements write to the output destination.
406 /// No other writes happen while this stream is alive.
407 ///
408 /// @note The document elements are indented up to 10 levels for readability.
409 /// The XML tree depth beyond 10 elements is printed at level 10.
410 class Stream {
411  public:
412  /// Constructs a document with XML header.
413  ///
414  /// @param[in] out The stream destination.
415  /// @param[in] indent Option to indent output for readability.
416  ///
417  /// @note This output file has clean error state.
418  explicit Stream(std::FILE* out, bool indent = true)
419  : indenter_(indent),
420  has_root_(false),
421  uncaught_exceptions_(std::uncaught_exceptions()),
422  out_(out) {
423  assert(!std::ferror(out) && "Unclean error state in output destination.");
424  out_ << "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n";
425  }
426 
427  /// @throws IOError The file write operation has failed.
428  ///
429  /// @post The exception is thrown only if no other exception is on flight.
430  ~Stream() noexcept(false) {
431  int err = std::ferror(out_.file());
432  if (err && (std::uncaught_exceptions() == uncaught_exceptions_))
433  SCRAM_THROW(IOError("FILE error on write")) << boost::errinfo_errno(err);
434  }
435 
436  /// Creates a root element for the document.
437  ///
438  /// @param[in] name The name for the root element.
439  ///
440  /// @returns XML stream element representing the document root.
441  ///
442  /// @pre The document is alive at least as long as the created root.
443  ///
444  /// @throws StreamError The document already has a root element,
445  /// or root element construction has failed.
446  StreamElement root(const char* name) {
447  if (has_root_)
448  throw StreamError("The XML stream document already has a root.");
449  StreamElement element(name, &indenter_, &out_);
450  has_root_ = true;
451  return element;
452  }
453 
454  private:
455  detail::Indenter indenter_; ///< The indentation manager for the document.
456  bool has_root_; ///< The document has constructed its root.
457  int uncaught_exceptions_; ///< The balance of exceptions.
458  detail::FileStream out_; ///< The output stream.
459 };
