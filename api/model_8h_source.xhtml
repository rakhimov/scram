43 /// This class represents a risk analysis model.

51  /// Container type identifier string for error messages.
52  static constexpr const char* kTypeString = "model";

54  /// Only Model is allowed to have an optional name,
55  /// while all other Elements require names.
56  /// An empty name is an error for Element class invariants as well.
57  /// This leads to a nasty magic string based optional name for a model.
58  static constexpr const char kDefaultName[] = "__unnamed-model__";

60  /// Creates a model container.
61  ///
62  /// @param[in] name The optional name for the model.
63  ///
64  /// @throws ValidityError The name is malformed.
65  explicit Model(std::string name = "");
66 
67  /// @returns true if the model name has not been set.
68  bool HasDefaultName() const { return Element::name() == kDefaultName; }
69 
70  /// @returns The model name or an empty string for the optional name.
71  const std::string& GetOptionalName() const {
72  static const std::string empty_name("");
73  return HasDefaultName() ? empty_name : Element::name();
74  }
75 
76  /// Sets the optional name of the model.
77  void SetOptionalName(std::string name = "") {
78  Element::name(name.empty() ? kDefaultName : std::move(name));
