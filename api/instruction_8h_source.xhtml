186 /// The base abstract class for instruction visitors.
187 class InstructionVisitor {
188  public:
189  virtual ~InstructionVisitor() = default;
190 
191  /// A set of required visitation functions for concrete visitors to implement.
192  /// @{
193  virtual void Visit(const SetHouseEvent*) = 0;
194  virtual void Visit(const CollectExpression*) = 0;
195  virtual void Visit(const CollectFormula*) = 0;
196  virtual void Visit(const Link*) = 0;
197  virtual void Visit(const IfThenElse* ite) {
198  if (ite->expression()->value()) {
199  ite->then_instruction()->Accept(this);
200  } else if (ite->else_instruction()) {
201  ite->else_instruction()->Accept(this);
202  }
203  }
204  virtual void Visit(const Block* block) {
205  for (const Instruction* instruction : block->instructions())
206  instruction->Accept(this);
207  }
208  virtual void Visit(const Rule* rule) {
209  for (const Instruction* instruction : rule->instructions())
210  instruction->Accept(this);
211  }
212  /// @}
213 };
