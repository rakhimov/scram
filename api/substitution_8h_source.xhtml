33 /// The general representation for
34 /// Delete Terms, Recovery Rules, and Exchange Events.
35 class Substitution : public Element {
36  public:
37  /// Type string for error messages.
38  static constexpr const char* kTypeString = "substitution";
39 
40  using Target = std::variant<BasicEvent*, bool>; ///< The target type.
41 
42  /// The "traditional" substitution types.
43  enum Type { kDeleteTerms, kRecoveryRule, kExchangeEvent };
44 
45  using Element::Element;
46 
47  /// @returns The formula hypothesis of the substitution.
48  ///
49  /// @pre The required hypothesis formula has been set.
50  const Formula& hypothesis() const {
51  assert(hypothesis_ && "Substitution hypothesis is not set.");
52  return *hypothesis_;
53  }
54 
55  /// Sets the substitution hypothesis formula.
56  ///
57  /// @param[in] formula Simple Boolean formula built over basic events only.
58  void hypothesis(std::unique_ptr<Formula> formula) {
59  assert(formula && "Cannot unset the hypothesis of substitution.");
60  hypothesis_ = std::move(formula);
61  }
62 
63  /// @returns The target of the substitution.
64  ///
65  /// @pre The target has been set.
66  const Target& target() const { return target_; }
67 
68  /// Sets the target of the substitution.
69  ///
70  /// @param[in] target_event Basic event or Boolean constant.
71  void target(Target target_event) { target_ = std::move(target_event); }
72 
73  /// @returns The source events of the substitution.
74  const std::vector<BasicEvent*>& source() const { return source_; }
75 
76  /// @returns true if the substitution is declarative.
77  bool declarative() const { return source_.empty(); }
78 
79  /// Adds a source event to the substitution container.
80  ///
81  /// @param[in] source_event The event to be replaced by the target event.
82  ///
83  /// @throws DuplicateElementError The source event is duplicate.
84  void Add(BasicEvent* source_event);
85 
86  /// Checks if the substitution is setup correctly.
87  ///
88  /// @pre The substitution has its hypothesis and target.
89  ///
90  /// @throws ValidityError Problems with the substitution setup.
91  ///
92  /// @note Non-declarative substitutions need to be validated further
93  /// for idempotency across substitutions before analysis.
94  void Validate() const;
95 
96  /// @returns The equivalent "traditional" substitution type if any.
97  ///
98  /// @pre The hypothesis, target, and source are all defined and valid.
99  std::optional<Type> type() const;
100 
101  private:
102  std::unique_ptr<Formula> hypothesis_; ///< The formula to be satisfied.
103  std::vector<BasicEvent*> source_; ///< The source events to be replaced.
104  Target target_; ///< The target event to replace the source events.
105 };
