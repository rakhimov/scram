39 /// Applies graph transformations consecutively.
40 ///
41 /// @param[in,out] graph The graph to be transformed.
42 /// @param[in] unary_op The first operation to be applied.
43 /// @param[in] unary_ops The rest of transformations to apply.
44 template <typename T, typename... Ts>
45 void Transform(Pdag* graph, T&& unary_op, Ts&&... unary_ops) noexcept {
46  if (graph->IsTrivial())
47  return;
48 
49  unary_op(graph);
50 
51  if constexpr (sizeof...(unary_ops))
52  Transform(graph, std::forward<Ts>(unary_ops)...);
53 }

559  /// Finds an optimal way of grouping options.
560  /// The goal of this function is
561  /// to maximize the effect of gate merging
562  /// or common argument factorization.
563  ///
564  /// After common arguments and parents are grouped,
565  /// the merging technique must find the most optimal strategy
566  /// to create new gates
567  /// that will represent the common arguments.
568  /// The strategy may favor modularity, size, or other parameters
569  /// of the new structure of the final graph.
570  /// The common elements within
571  /// the groups of common parents and common arguments
572  /// create the biggest challenge for finding the optimal solution.
573  /// For example,
574  /// {
