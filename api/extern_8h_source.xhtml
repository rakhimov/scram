41 /// The MEF construct to extend expressions with external libraries.
42 /// This class dynamically loads and manages libraries.
43 /// It supports only very basic interface for C function lookup with its symbol.
44 class ExternLibrary : public Element, public Usage {
45  public:
46  /// Type string for error messages.
47  static constexpr const char* kTypeString = "extern library";
48 
49  /// @copydoc Element::Element
50  ///
51  /// @param[in] lib_path The library path with its name.
52  /// @param[in] reference_dir The reference directory for relative paths.
53  /// @param[in] system Search for the library in system paths.
54  /// @param[in] decorate Decorate the library name with prefix and suffix.
55  ///
56  /// @throws ValidityError The library path is invalid.
57  /// @throws DLError The library cannot be found.
58  ExternLibrary(std::string name, std::string lib_path,
59  const boost::filesystem::path& reference_dir, bool system,
60  bool decorate);
61 
62  /// @tparam F The C free function type.
63  ///
64  /// @param[in] symbol The function symbol in the library.
65  ///
66  /// @returns The function pointer resolved from the symbol.
67  ///
68  /// @throws DLError The symbol is not in the library.
69  template <typename F>
70  std::enable_if_t<std::is_function_v<F>, std::add_pointer_t<F>>
71  get(const std::string& symbol) const {
72  try {
73  return lib_handle_.get<F>(symbol);
74  } catch (const boost::system::system_error& err) {
75  SCRAM_THROW(DLError(err.what()))
76  << errinfo_value(symbol)
77  << boost::errinfo_nested_exception(boost::current_exception());
78  }
79  }
80 
81  private:
82  boost::dll::shared_library lib_handle_; ///< Shared Library abstraction.
83 };

118 /// Extern function abstraction to be referenced by expressions.
119 ///
120 /// @tparam R Numeric return type.
121 /// @tparam Args Numeric argument types.
122 ///
123 /// @pre The source dynamic library is loaded as long as this function lives.
124 template <typename R, typename... Args>
125 class ExternFunction : public ExternFunctionBase {
126  static_assert(std::is_arithmetic_v<R>, "Numeric type functions only.");
127 
128  using Pointer = R (*)(Args...); ///< The function pointer type.
129 
130  public:
131  /// Loads a function from a library for further usage.
132  ///
133  /// @copydoc Element::Element
134  ///
135  /// @param[in] symbol The symbol name for the function in the library.
136  /// @param[in] library The dynamic library to lookup the function.
137  ///
138  /// @throws DLError There is no such symbol in the library.
139  ExternFunction(std::string name, const std::string& symbol,
140  const ExternLibrary& library)
141  : ExternFunctionBase(std::move(name)),
142  fptr_(library.get<R(Args...)>(symbol)) {}
143 
144  /// Calls the library function with the given numeric arguments.
145  R operator()(Args... args) const noexcept { return fptr_(args...); }
146 
147  /// @copydoc ExternFunction<void>::apply
148  std::unique_ptr<Expression>
149  apply(std::vector<Expression*> args) const override;
150 
151  private:
152  const Pointer fptr_; ///< The pointer to the extern function in a library.
153 };

155 /// Expression evaluating an extern function with expression arguments.
156 ///
157 /// @tparam R Numeric return type.
158 /// @tparam Args Numeric argument types.
159 template <typename R, typename... Args>
160 class ExternExpression
161  : public ExpressionFormula<ExternExpression<R, Args...>> {
162  public:
163  /// @param[in] extern_function The library function.
164  /// @param[in] args The argument expression for the function.
165  ///
166  /// @throws ValidityError The number of arguments is invalid.
167  explicit ExternExpression(const ExternFunction<R, Args...>* extern_function,
168  std::vector<Expression*> args)
169  : ExpressionFormula<ExternExpression>(std::move(args)),
170  extern_function_(*extern_function) {
171  if (Expression::args().size() != sizeof...(Args))
172  SCRAM_THROW(
173  ValidityError("The number of function arguments does not match."));
174  }
175 
176  /// Computes the extern function with the given evaluator for arguments.
177  template <typename F>
178  double Compute(F&& eval) noexcept {
179  return Marshal(std::forward<F>(eval),
180  std::make_index_sequence<sizeof...(Args)>());
181  }
182 
183  private:
184  /// Evaluates the argument expressions and marshals the result to function.
185  /// Marshaller of expressions to extern function calls.
186  ///
187  /// @tparam F The expression evaluator type.
188  /// @tparam Is The index sequence for the arguments vector.
189  ///
190  /// @param[in] eval The evaluator of the expressions.
191  ///
192  /// @returns The result of the function call.
193  ///
194  /// @pre The number of arguments is exactly the same at runtime.
195  template <typename F, std::size_t... Is>
196  double Marshal(F&& eval, std::index_sequence<Is...>) noexcept {
197  return extern_function_(eval(Expression::args()[Is])...);
198  }
199 
200  const ExternFunction<R, Args...>& extern_function_; ///< The source function.
201 };
