63 /// Gets a number from an XML value.
64 ///
65 /// @tparam T Numeric type.
66 ///
67 /// @param[in] value The non-empty value string.
68 ///
69 /// @returns The interpreted value.
70 ///
71 /// @throws ValidityError The interpretation is unsuccessful.
72 template <typename T>
73 std::enable_if_t<std::is_arithmetic_v<T>, T> to(const std::string_view& value) {
74  if constexpr (std::is_same_v<T, int>) {
75  char* end_char = nullptr;
76  std::int64_t ret = std::strtoll(value.data(), &end_char, 10);
77  int len = end_char - value.data();
78  if (len != value.size() || ret > std::numeric_limits<int>::max() ||
79  ret < std::numeric_limits<int>::min()) {
80  SCRAM_THROW(ValidityError("Failed to interpret value to int"))
81  << errinfo_value(std::string(value));
82  }
83  return ret;
84 
85  } else if constexpr (std::is_same_v<T, double>) { // NOLINT
86  char* end_char = nullptr;
87  double ret = std::strtod(value.data(), &end_char);
88  int len = end_char - value.data();
89  if (len != value.size() || ret == HUGE_VAL || ret == -HUGE_VAL) {
90  SCRAM_THROW(ValidityError("Failed to interpret value to double"))
91  << errinfo_value(std::string(value));
92  }
93  return ret;
94 
95  } else {
96  static_assert(std::is_same_v<T, bool>, "Only default numeric types.");
97 
98  if (value == "true" || value == "1")
99  return true;
100  if (value == "false" || value == "0")
101  return false;
102  SCRAM_THROW(ValidityError("Failed to interpret value to bool"))
103  << errinfo_value(std::string(value));
104  }
105 }

129 /// Removes leading and trailing space characters from XML value string.
130 ///
131 /// @param[in] text The text in XML attribute or text nodes.
132 ///
133 /// @returns View to the trimmed substring.
134 ///
135 /// @pre The string is normalized by the XML parser.
136 inline std::string_view trim(const std::string_view& text) noexcept {
137  auto pos_first = text.find_first_not_of(' ');
138  if (pos_first == std::string_view::npos)
139  return {};
140 
141  auto pos_last = text.find_last_not_of(' ');
142  auto len = pos_last - pos_first + 1;
143 
144  return std::string_view(text.data() + pos_first, len);
145 }

147 /// Gets the last XML error converted from the library error codes.
148 ///
149 /// @tparam T The SCRAM error type to convert XML error into.
150 ///
151 /// @param[in] xml_error The error to translate.
152 /// nullptr to retrieve the latest error in the library.
153 ///
154 /// @returns The exception object to be thrown.
155 template <typename T>
156 T GetError(xmlErrorPtr xml_error = nullptr) {
157  if (!xml_error)
158  xml_error = xmlGetLastError();
159  assert(xml_error && "No XML error is available.");
160  T throw_error(xml_error->message);
161  if (xml_error->file)
162  throw_error << boost::errinfo_file_name(xml_error->file);
163  if (xml_error->line)
164  throw_error << boost::errinfo_at_line(xml_error->line);
165  return throw_error;
166 }

279  /// Retrieves the XML element's attribute values.
280  ///
281  /// @param[in] name The name of the requested attribute.
282  ///
283  /// @returns The attribute value or
284  /// empty string if no attribute (optional attribute).
285  ///
286  /// @pre XML attributes never contain empty strings.
287  /// @pre XML attribute values are simple texts w/o DTD processing.
288  std::string_view attribute(const char* name) const {
289  const xmlAttr* property = xmlHasProp(to_node(), detail::to_utf8(name));
290  if (!property)
291  return {};
292  const xmlNode* text_node = property->children;
293  assert(text_node && text_node->type == XML_TEXT_NODE);
294  assert(text_node->content);
295  return detail::trim(detail::from_utf8(text_node->content));
296  }

298  /// Generic attribute value extraction following XML data types.
299  ///
300  /// @tparam T The attribute value type (numeric).
301  ///
302  /// @param[in] name The name of the attribute.
303  ///
304  /// @returns The value of type T interpreted from attribute value.
305  /// None if the attribute doesn't exists (optional).
306  ///
307  /// @throws ValidityError Casting is unsuccessful.
308  template <typename T>
309  std::enable_if_t<std::is_arithmetic_v<T>, std::optional<T>>
310  attribute(const char* name) const {
311  std::string_view value = attribute(name);
312  if (value.empty())
313  return {};
314  try {
315  return detail::to<T>(value);
316  } catch (ValidityError& err) {
317  err << errinfo_element(std::string(Element::name()))
318  << errinfo_attribute(name) << boost::errinfo_at_line(line())
319  << boost::errinfo_file_name(filename());
320  throw;
321  }
322  }
