35 /// Component is for logical grouping of events, gates, and other components.
36 class Component
37  : public Element,
38  public Role,
39  public Composite<Container<Component, Component, true, false>,
40  Container<Component, BasicEvent, false, false>,
41  Container<Component, HouseEvent, false, false>,
42  Container<Component, Gate, false, false>,
43  Container<Component, Parameter, false, false>,
44  Container<Component, CcfGroup, false, false>> {
45  public:
46  /// Type identifier string for error messages.
47  static constexpr const char* kTypeString = "component/fault tree";
48 
49  /// Constructs a component assuming
50  /// that it exists within some fault tree.
51  /// The public or private role of a component is not
52  /// for the components itself,
53  /// but for the events and parameters of the component.
54  /// Component name is not meant to be public;
55  /// however, it must be unique within the parent fault tree or component.
56  ///
57  /// @param[in] name The name identifier for the component.
58  /// @param[in] base_path The series of containers to get this container.
59  /// @param[in] role The default role for container members.
60  ///
61  /// @throws LogicError The name is empty.
62  /// @throws ValidityError The name or reference paths are malformed.
63  explicit Component(std::string name, std::string base_path = "",
64  RoleSpecifier role = RoleSpecifier::kPublic);
65 
66  virtual ~Component() = default;
67 
68  /// @returns The table ranges of component elements of specific kind
69  /// with element original names as keys.
70  /// @{
71  auto gates() const { return table<Gate>(); }
72  auto basic_events() const { return table<BasicEvent>(); }
73  auto house_events() const { return table<HouseEvent>(); }
74  auto parameters() const { return table<Parameter>(); }
75  auto ccf_groups() const { return table<CcfGroup>(); }
76  auto components() const { return table<Component>(); }
77  /// @}
78 
79  using Composite::Add;
80  using Composite::Remove;
81 
82  /// Adds MEF constructs into this component container.
83  ///
84  /// @param[in] element The element to be added to the container.
85  ///
86  /// @throws DuplicateElementError The element is already in this container.
87  ///
88  /// @{
89  void Add(Gate* element) { AddEvent(element); }
90  void Add(BasicEvent* element) { AddEvent(element); }
91  void Add(HouseEvent* element) { AddEvent(element); }

102 
103  private:

119 /// Fault tree representation as a container of
120 /// gates, basic and house events, and other information.
121 /// Additional functionality of a fault tree includes
122 /// detection of top events.
123 class FaultTree : public Component {
124  public:
125  /// Type identifier string for error messages.
126  static constexpr const char* kTypeString = "fault tree";
127 
128  /// The main constructor of the Fault Tree.
129  /// Fault trees are assumed to be public and belong to the root model.
130  ///
131  /// @param[in] name The name identifier of this fault tree.
132  explicit FaultTree(const std::string& name);
133 
134  /// @returns The collected top events of this fault tree.
135  const std::vector<const Gate*>& top_events() const { return top_events_; }
136 
137  /// Collects top event gates in this fault tree with components.
138  /// This function is essential to guess the analysis targets
139  /// if the user does not supply any.
140  /// If the structure of the fault tree changes,
141  /// this function must be called again to update the top events.
142  ///
143  /// @pre Gate marks are clear.
144  void CollectTopEvents();
145 
146  private:
147  /// Recursively marks descendant gates
148  /// with an unspecified-but-non-clear mark.
149  /// These gates belong to this fault tree only.
150  ///
151  /// @param[in,out] gate The ancestor gate.
152  /// @param[in] gates Gates belonging to the whole fault tree with components.
153  void MarkNonTopGates(Gate* gate, const std::unordered_set<Gate*>& gates);
154 
155  std::vector<const Gate*> top_events_; ///< Top events of this fault tree.
156 };
