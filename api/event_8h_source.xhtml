78 /// Representation of a basic event in a fault tree.
79 class BasicEvent : public Event {
80  public:
81  static constexpr const char* kTypeString = "basic event"; ///< In errors.
82 
83  using Event::Event;
84 
85  virtual ~BasicEvent() = default;
86 
87  /// @returns true if the probability expression is set.
88  bool HasExpression() const { return expression_ != nullptr; }
89 
90  /// Sets the expression of this basic event.
91  ///
92  /// @param[in] expression The expression to describe this event.
93  /// nullptr to remove unset the expression.
94  void expression(Expression* expression) { expression_ = expression; }
95 
96  /// @returns The previously set expression for analysis purposes.
97  ///
98  /// @pre The expression has been set.
99  Expression& expression() const {
100  assert(expression_ && "The basic event's expression is not set.");
101  return *expression_;
102  }
103 
104  /// @returns The mean probability of this basic event.
105  ///
106  /// @pre The expression has been set.
107  ///
108  /// @note The user of this function should make sure
109  /// that the returned value is acceptable for calculations.
110  double p() const noexcept {
111  assert(expression_ && "The basic event's expression is not set.");
112  return expression_->value();
113  }
114 
115  /// Validates the probability expressions for the primary event.
116  ///
117  /// @pre The probability expression is set.
118  ///
119  /// @throws DomainError The expression for the basic event is invalid.
120  void Validate() const;
121 
122  /// Indicates if this basic event has been set to be in a CCF group.
123  ///
124  /// @returns true if in a CCF group.
125  bool HasCcf() const { return ccf_gate_ != nullptr; }
126 
127  /// @returns CCF group gate representing this basic event.
128  const Gate& ccf_gate() const {
129  assert(ccf_gate_);
130  return *ccf_gate_;
131  }
132 
133  /// Sets the common cause failure group gate
134  /// that can represent this basic event
135  /// in analysis with common cause information.
136  /// This information is expected to be provided by
137  /// CCF group application.
138  ///
139  /// @param[in] gate CCF group gate.
140  void ccf_gate(std::unique_ptr<Gate> gate) {
141  assert(!ccf_gate_);
142  ccf_gate_ = std::move(gate);
143  }
144 
145  private:
146  /// Expression that describes this basic event
147  /// and provides numerical values for probability calculations.
148  Expression* expression_ = nullptr;
149 
150  /// If this basic event is in a common cause group,
151  /// CCF gate can serve as a replacement for the basic event
152  /// for common cause analysis.
153  std::unique_ptr<Gate> ccf_gate_;
154 };

197 /// Logical connectives for formulas.
198 /// The ordering is the same as analysis connectives in the PDAG.
199 enum Connective : std::uint8_t {
200  kAnd = 0,
201  kOr,
202  kAtleast, ///< Combination, K/N, atleast, or Vote gate representation.
203  kXor, ///< Exclusive OR gate with two inputs only.
204  kNot, ///< Boolean negation.
205  kNand, ///< Not AND.
206  kNor, ///< Not OR.
207  kNull, ///< Single argument pass-through without logic.
208 
209  // Rarely used connectives specific to the MEF.
210  kIff, ///< Equality with two inputs only.
211  kImply, ///< Implication with two inputs only.
212  kCardinality ///< General quantifier of events.
213 };
