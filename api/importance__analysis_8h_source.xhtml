145 template <class Calculator>
146 double ImportanceAnalyzer<Calculator>::CalculateMif(int index) noexcept {
147  index += Pdag::kVariableStartIndex;
148  auto p_conditional = [index, this](bool state) {
149  p_vars_[index] = state;
150  return static_cast<ProbabilityAnalyzer<Calculator>*>(prob_analyzer())
151  ->CalculateTotalProbability(p_vars_);
152  };
153  double p_store = p_vars_[index]; // Save the original value for restoring.
154  double mif = p_conditional(true) - p_conditional(false);
155  p_vars_[index] = p_store; // Restore the probability for next calculation.
156  return mif;
157 }
