42 /// A basic event that represents a multiple failure of
43 /// a group of events due to a common cause.
44 /// This event is generated out of a common cause group.
45 /// This class is a helper to report correctly the CCF events.
46 class CcfEvent : public BasicEvent {
47  public:
48  /// Constructs CCF event with specific name
49  /// that is used for internal purposes.
50  /// This name is formatted with the CcfGroup.
51  /// The creator CCF group and the member events of this specific CCF event
52  /// are saved for reporting.
53  ///
54  /// @param[in] members The members that this CCF event
55  /// represents as multiple failure.
56  /// @param[in] ccf_group The CCF group that created this event.
57  CcfEvent(std::vector<Gate*> members, const CcfGroup* ccf_group);
58 
59  /// @returns The CCF group that created this CCF event.
60  const CcfGroup& ccf_group() const { return ccf_group_; }
61 
62  /// @returns Members of this CCF event.
63  /// The members also own this CCF event through parentship.
64  const std::vector<Gate*>& members() const { return members_; }
65 
66  private:
67  /// Creates a mangled name
68  /// that is specific to CCF events and unique per model.
69  ///
70  /// @param[in] members The members that this CCF event represents.
71  ///
72  /// @returns The name string valid only for internal uses.
73  static std::string MakeName(const std::vector<Gate*>& members);
74 
75  const CcfGroup& ccf_group_; ///< The originating CCF group.
76  std::vector<Gate*> members_; ///< Member parent gates of this CCF event.
77 };

92  /// Adds a basic event into this CCF group.
93  /// This function asserts that each basic event has unique string id.
94  ///
95  /// @param[in] basic_event A member basic event.

103  /// Adds the distribution that describes the probability of
104  /// basic events in this CCF group.
105  /// All basic events should be added as members
106  /// before defining their probabilities.
107  /// No more basic events can be added after this function.
108  ///
109  /// @param[in] distr The probability distribution of this group.
110  ///
111  /// @throws ValidityError Not enough members.
112  /// @throws LogicError The distribution has already been defined.
113  void AddDistribution(Expression* distr);
114 
115  /// Adds a CCF factor for the specified model.
116  /// All basic events should be added as members
117  /// before defining the CCF factors.
118  /// No more basic events can be added after this function.
119  ///
120  /// @param[in] factor A factor for the CCF model.
121  /// @param[in] level The level of the passed factor.
122  ///
123  /// @throws ValidityError The level is invalid.
124  /// @throws ValidityError The factor for the level already exists.
125  /// @throws LogicError The level is not positive,
126  /// or the CCF group members are undefined.
127  void AddFactor(Expression* factor, std::optional<int> level = {});
128 
129  /// Validates the setup for the CCF model and group.
130  /// Checks if the provided distribution is between 0 and 1.
131  ///
132  /// This check must be performed before validating basic events

134  /// to give more precise error messages.
135  ///
136  /// @throws DomainError The numeric values are invalid.
