145  /// Registers the visit time for this node upon graph traversal.
146  /// This information can be used to detect dependencies.
147  ///
148  /// @param[in] time The current visit time of this node.
149  /// It must be positive.
150  ///
151  /// @returns true if this node was previously visited.
152  /// @returns false if this is visited and re-visited only once.
153  bool Visit(int time) {
154  assert(time > 0);
155  if (!visits_[0]) {
156  visits_[0] = time;
157  } else if (!visits_[1]) {
158  visits_[1] = time;
159  } else {
160  visits_[2] = time;
161  return true;
162  }
163  return false;
164  }
165 
166  /// @returns The time when this node was first encountered or entered.
167  /// @returns 0 if no enter time is registered.
168  int EnterTime() const { return visits_[0]; }
169 
170  /// @returns The exit time upon traversal of the graph.
171  /// @returns 0 if no exit time is registered.
172  int ExitTime() const { return visits_[1]; }
173 
174  /// @returns The last time this node was visited.
175  /// @returns 0 if no last time is registered.
176  int LastVisit() const { return visits_[2] ? visits_[2] : visits_[1]; }
177 
178  /// @returns The minimum time of the visit.
179  /// @returns 0 if no time is registered.
180  virtual int min_time() const { return visits_[0]; }
181 
182  /// @returns The maximum time of the visit.
183  /// @returns 0 if no time is registered.
184  virtual int max_time() const { return LastVisit(); }
185 
186  /// @returns false if this node was only visited once upon graph traversal.
187  /// @returns true if this node was revisited at least one more time.
188  bool Revisited() const { return visits_[2]; }
189 
190  /// @returns true if this node was visited at least once.
191  /// @returns false if this node was never visited upon traversal.
192  bool Visited() const { return visits_[0]; }
193 
194  /// Clears all the visit information. Resets the visit times to 0s.
195  void ClearVisits() { std::fill_n(visits_, 3, 0); }
196 
197  /// @returns The positive count of this node.
198  int pos_count() const { return pos_count_; }
199 
200  /// @returns The negative count of this node.
201  int neg_count() const { return neg_count_; }
202 
203  /// Increases the count of this node.
204  ///
205  /// @param[in] positive Indication of a positive node.
206  void AddCount(bool positive) { positive ? ++pos_count_ : ++neg_count_; }
207 
208  /// Resets positive and negative counts of this node.
209  void ResetCount() {
210  pos_count_ = 0;
211  neg_count_ = 0;
212  }
213 
214  private:
215  int index_; ///< Index of this node.
216  int order_; ///< Ordering of nodes in the graph.
217  int visits_[3]; ///< Traversal array with first, second, and last visits.
218  int opti_value_; ///< Failure propagation optimization value.
219  int pos_count_; ///< The number of occurrences as a positive node.
220  int neg_count_; ///< The number of occurrences as a negative node.
221  Pdag& graph_; ///< The host graph for the node.
222 };

251 /// Boolean connectives of gates
252 /// for representation, preprocessing, and analysis purposes.
253 /// The connective defines a type and logic of a gate.
254 ///
255 /// @warning If a new connective is added,
256 /// all the preprocessing and PDAG algorithms
257 /// must be reviewed and updated.
258 /// The algorithms may assume
259 /// for performance and simplicity reasons
260 /// that these are the only kinds of connectives possible.
261 enum Connective : std::uint8_t {
262  kAnd = 0, ///< AND gate.
263  kOr, ///< OR gate.
264  kAtleast, ///< Combination, K/N, or Vote gate representation.
265  kXor, ///< Exclusive OR gate with two inputs.
266  kNot, ///< Boolean negation.
267  kNand, ///< NAND gate.
268  kNor, ///< NOR gate.
269  kNull ///< Special pass-through or NULL gate. This is not NULL set.
270 };

513  /// Adds an argument node to this gate.
514  ///
515  /// Before adding the argument,
516  /// the existing arguments are checked for complements and duplicates.
517  /// If there is a complement,
518  /// the gate may change its state (erasing its arguments) or type.
519  ///
520  /// The duplicates are handled according to the logic of the gate.
521  /// The caller must be aware of possible changes
522  /// due to the logic of the gate.
523  ///
524  /// @tparam T The type of the argument node.
525  ///
526  /// @param[in] index A positive or negative index of an argument.
527  /// @param[in] arg A pointer to the argument node.
528  ///
529  /// @warning The function does not indicate invalid state.
530  /// For example, a second argument for NOT or NULL type gates
531  /// is not going to be reported in any way.
532  /// @warning This function does not indicate error
533  /// for future additions
534  /// in case the state is nulled or becomes unity.
535  /// @warning Duplicate arguments may change the type and state of the gate.
536  /// Depending on the logic of the gate,
537  /// new gates may be introduced
538  /// instead of the existing arguments.
539  /// @warning Complex logic gates like ATLEAST and XOR
540  /// are handled specially
541  /// if the argument is duplicate.
542  /// The caller must be very cautious of
543  /// the side effects of the manipulations.
544  template <class T>
545  void AddArg(int index, const std::shared_ptr<T>& arg) noexcept {
546  assert(index);
547  assert(std::abs(index) == arg->index());
548  assert(!constant_);
549  assert(!((type_ == kNot || type_ == kNull) && !args_.empty()));
550  assert(!(type_ == kXor && args_.size() > 1));
551  assert(min_number_ >= 0);
552 
553  if (args_.count(index))
554  return ProcessDuplicateArg(index);
555  if (args_.count(-index))
556  return ProcessComplementArg(index);
557 
558  args_.insert(index);
559  mutable_args<T>().data().emplace_back(index, arg);
560  arg->AddParent(shared_from_this());
561  }

1198  node->opti_value(0);
