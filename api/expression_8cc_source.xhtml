118 void EnsureWithin(Expression* expression, const Interval& interval,
119  const char* type) {
120  double arg_value = expression->value();

122  std::stringstream ss;

124  SCRAM_THROW(DomainError(ss.str()))

127  Interval arg_interval = expression->interval();
128  if (!boost::icl::within(arg_interval, interval)) {
129  std::stringstream ss;
