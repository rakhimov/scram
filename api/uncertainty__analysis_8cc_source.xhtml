73 void UncertaintyAnalysis::SampleExpressions(
74  const std::vector<std::pair<int, mef::Expression&>>& deviate_expressions,
75  Pdag::IndexMap<double>* p_vars) noexcept {
76  // Reset distributions.
77  for (const auto& expression : deviate_expressions)
78  expression.second.Reset();
79 
80  // Sample all expressions with distributions.
81  for (const auto& expression : deviate_expressions) {
82  double prob = expression.second.Sample();
83  (*p_vars)[expression.first] = prob > 1 ? 1 : prob < 0 ? 0 : prob;
84  }
85 }

87 void UncertaintyAnalysis::CalculateStatistics(
88  const std::vector<double>& samples) noexcept {
89  using namespace boost; // NOLINT
90  using namespace boost::accumulators; // NOLINT
91  using histogram_type =
92  iterator_range<std::vector<std::pair<double, double>>::iterator>;
93  quantiles_.clear();
94  int num_quantiles = Analysis::settings().num_quantiles();
95  double delta = 1.0 / num_quantiles;
96  for (int i = 0; i < num_quantiles; ++i) {
97  quantiles_.push_back(delta * (i + 1));
98  }
99  int num_trials = Analysis::settings().num_trials();
100  accumulator_set<double, stats<tag::mean, tag::variance, tag::density,
101  tag::extended_p_square_quantile>>
102  acc(tag::density::num_bins = Analysis::settings().num_bins(),
103  tag::density::cache_size = num_trials,
104  extended_p_square_probabilities = quantiles_);
105  for (double sample : samples) {
106  acc(sample);
107  }
108  histogram_type hist = density(acc);
109  for (int i = 1; i < hist.size(); i++) {
110  distribution_.push_back(hist[i]);
111  }
112  mean_ = boost::accumulators::mean(acc);
113  sigma_ = std::sqrt(num_trials * variance(acc) / (num_trials - 1));
114  error_factor_ = std::exp(1.96 * sigma_);
115  confidence_interval_.first = mean_ - sigma_ * 1.96 / std::sqrt(num_trials);
116  confidence_interval_.second = mean_ + sigma_ * 1.96 / std::sqrt(num_trials);
117 
118  for (int i = 0; i < num_quantiles; ++i) {
119  quantiles_[i] = quantile(acc, quantile_probability = quantiles_[i]);
120  }
121 }
