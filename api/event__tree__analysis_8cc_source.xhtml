37 /// Clones the formula by applying the set-instructions.
38 ///
39 /// @param[in] formula The formula to be deep-cloned.
40 /// @param[in] set_instructions The set instructions to change arguments.
41 /// @param[in] clones The storage container for newly created clones.
42 ///
43 /// @returns The deep-copy of the argument formula with new (changed) arguments.
44 std::unique_ptr<mef::Formula>
45 Clone(const mef::Formula& formula,
46  const std::unordered_map<std::string, bool>& set_instructions,
47  std::vector<std::unique_ptr<mef::Event>>* clones) noexcept {
48  struct {
49  mef::Formula::ArgEvent operator()(mef::BasicEvent* arg) { return arg; }
50  mef::Formula::ArgEvent operator()(mef::HouseEvent* arg) {
51  if (auto it = ext::find(set_house, arg->id())) {
52  if (it->second == arg->state())
53  return arg;
54  auto clone = std::make_unique<mef::HouseEvent>(
55  arg->name(), "__clone__." + arg->id(),
56  mef::RoleSpecifier::kPrivate);
57  clone->state(it->second);
58  auto* ptr = clone.get();
59  event_register->emplace_back(std::move(clone));
60  return ptr;
61  }
62  return arg;
63  }
64  mef::Formula::ArgEvent operator()(mef::Gate* arg) {
65  if (set_house.empty())
66  return arg;
67  auto clone = std::make_unique<mef::Gate>(
68  arg->name(), "__clone__." + arg->id(), mef::RoleSpecifier::kPrivate);
69  clone->formula(Clone(arg->formula(), set_house, event_register));
70  auto* ptr = clone.get();
71  event_register->emplace_back(std::move(clone));
72  return ptr;
73  }
74 
75  const std::unordered_map<std::string, bool>& set_house;
76  std::vector<std::unique_ptr<mef::Event>>* event_register;
77  } cloner{set_instructions, clones};
78 
79  mef::Formula::ArgSet arg_set;
80  for (const mef::Formula::Arg& arg : formula.args())
81  arg_set.Add(std::visit(cloner, arg.event), arg.complement);
82 
83  return std::make_unique<mef::Formula>(
84  formula.connective(), std::move(arg_set), formula.min_number(),
85  formula.max_number());
86 }

96 void EventTreeAnalysis::Analyze() noexcept {
97  assert(initiating_event_.event_tree());
98  int formula_id = 0; // Enumeration of collected formulas turned into gates.
99  // Creates an internal gate representing the formula.
100  auto make_gate = [&formula_id, this](mef::FormulaPtr formula) {
101  std::string gate_name = "___" + initiating_event_.name() + "__formula_" +
102  std::to_string(formula_id++) + "__";
103  auto gate = std::make_unique<mef::Gate>(gate_name);
104  gate->formula(std::move(formula));
105  auto* address = gate.get();
106  events_.emplace_back(std::move(gate));
107  return address;
108  };
109 
110  SequenceCollector collector{initiating_event_, *context_};
111  CollectSequences(initiating_event_.event_tree()->initial_state(), &collector);
112  for (auto& sequence : collector.sequences) {
113  auto gate = std::make_unique<mef::Gate>("__" + sequence.first->name());
114  std::vector<mef::FormulaPtr> gate_formulas;
115  std::vector<mef::Expression*> arg_expressions;
116  for (PathCollector& path_collector : sequence.second) {
117  if (path_collector.formulas.size() == 1) {
118  gate_formulas.push_back(std::move(path_collector.formulas.front()));
119  } else if (path_collector.formulas.size() > 1) {
120  mef::Formula::ArgSet arg_set;
121  for (mef::FormulaPtr& arg_formula : path_collector.formulas)
122  arg_set.Add(make_gate(std::move(arg_formula)));
123 
124  gate_formulas.push_back(
125  std::make_unique<mef::Formula>(mef::kAnd, std::move(arg_set)));
126  }
127  if (path_collector.expressions.size() == 1) {
128  arg_expressions.push_back(path_collector.expressions.front());
129  } else if (path_collector.expressions.size() > 1) {
130  expressions_.push_back(
131  std::make_unique<mef::Mul>(std::move(path_collector.expressions)));
132  arg_expressions.push_back(expressions_.back().get());
133  }
134  }
135  assert(gate_formulas.empty() || arg_expressions.empty());
136  bool is_expression_only = !arg_expressions.empty();
137  if (gate_formulas.size() == 1) {
138  gate->formula(std::move(gate_formulas.front()));
139  } else if (gate_formulas.size() > 1) {
140  mef::Formula::ArgSet arg_set;
141  for (mef::FormulaPtr& arg_formula : gate_formulas)
142  arg_set.Add(make_gate(std::move(arg_formula)));
143 
144  gate->formula(
145  std::make_unique<mef::Formula>(mef::kOr, std::move(arg_set)));
146  } else if (!arg_expressions.empty()) {
147  auto event =
148  std::make_unique<mef::BasicEvent>("__" + sequence.first->name());
149  if (arg_expressions.size() == 1) {
150  event->expression(arg_expressions.front());
151  } else if (arg_expressions.size() > 1) {
152  expressions_.push_back(
153  std::make_unique<mef::Add>(std::move(arg_expressions)));
154  event->expression(expressions_.back().get());
155  }
156  gate->formula(std::make_unique<mef::Formula>(
157  mef::kNull, mef::Formula::ArgSet{event.get()}));
158  events_.push_back(std::move(event));
159  } else {
160  gate->formula(std::make_unique<mef::Formula>(
161  mef::kNull, mef::Formula::ArgSet{&mef::HouseEvent::kTrue}));
162  }
163  sequences_.push_back(
164  {*sequence.first, std::move(gate), is_expression_only});
165  }

168 void EventTreeAnalysis::CollectSequences(const mef::Branch& initial_state,
169  SequenceCollector* result) noexcept {
170  struct Collector {
171  class Visitor : public mef::InstructionVisitor {
172  public:
173  explicit Visitor(Collector* collector) : collector_(*collector) {}
174 
175  void Visit(const mef::SetHouseEvent* house_event) override {
176  collector_.path_collector_.set_instructions[house_event->name()] =
177  house_event->state();
178  }
179 
180  void Visit(const mef::Link* link) override {
181  is_linked_ = true;
182  Collector continue_connector(collector_);
183  auto save = std::move(collector_.result_->context.functional_events);
184  continue_connector(&link->event_tree().initial_state());
185  collector_.result_->context.functional_events = std::move(save);
186  }
187 
188  void Visit(const mef::CollectFormula* collect_formula) override {
189  // clang-format off
190  collector_.path_collector_.formulas.push_back(
191  core::Clone(collect_formula->formula(),
192  collector_.path_collector_.set_instructions,
193  collector_.clones_));
194  // clang-format on
195  }
196 
197  void Visit(const mef::CollectExpression* collect_expression) override {
198  collector_.path_collector_.expressions.push_back(
199  &collect_expression->expression());
200  }
201 
202  bool is_linked() const { return is_linked_; }
203 
204  private:
205  Collector& collector_;
206  bool is_linked_ = false; /// Indicate that sequences not be registered.
207  };
208 
209  void operator()(const mef::Sequence* sequence) {
210  Visitor visitor(this);
211  for (const mef::Instruction* instruction : sequence->instructions())
212  instruction->Accept(&visitor);
213  if (!visitor.is_linked())
214  result_->sequences[sequence].push_back(std::move(path_collector_));
215  }
216 
217  void operator()(const mef::Fork* fork) const {
218  const std::string& name = fork->functional_event().name();
219  assert(result_->context.functional_events.count(name) == false);
220  std::string& state = result_->context.functional_events[name];
221  assert(state.empty());
222  for (const mef::Path& fork_path : fork->paths()) {
223  state = fork_path.state();
224  // clang-format off
225  Collector(*this)(&fork_path); // NOLINT(runtime/explicit)
226  // clang-format on
227  }
228  result_->context.functional_events.erase(name);
229  }
230 
231  void operator()(const mef::Branch* branch) {
232  Visitor visitor(this);
233  for (const mef::Instruction* instruction : branch->instructions())
234  instruction->Accept(&visitor);
235 
236  std::visit(*this, branch->target());
237  }
238 
239  SequenceCollector* result_;
240  std::vector<std::unique_ptr<mef::Event>>* clones_;
241  PathCollector path_collector_;
242  };
243  context_->functional_events.clear();
244  context_->initiating_event = initiating_event_.name();
245  Collector{result, &events_}(&initial_state); // NOLINT(whitespace/braces)
246 }
