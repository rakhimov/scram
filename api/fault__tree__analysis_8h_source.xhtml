53 /// Collection of unique literals.
54 class Product {
55  /// Converter of indices in products to Literals.
56  struct LiteralExtractor {
57  /// @param[in] index The index in a result product.
58  ///
59  /// @returns Literal representing the event with the index.
60  Literal operator()(int index) const {
61  return {index < 0, *graph.basic_events()[std::abs(index)]};
62  }
63  const Pdag& graph; ///< The host graph.
64  };
65 
66  public:
67  /// Initializes product literals.
68  ///
69  /// @param[in] data The underlying set.
70  /// @param[in] graph The graph with indices to events map.
71  Product(const std::vector<int>& data, const Pdag& graph) noexcept
72  : data_(data), graph_(graph) {}
73 
74  /// @returns true for unity product with no literals.
75  bool empty() const { return data_.empty(); }
76 
77  /// @returns The number of literals in the product.
78  int size() const { return data_.size(); }
79 
80  /// @returns The order of the product.
81  ///
82  /// @note An empty set indicates the Base/Unity set.
83  int order() const { return empty() ? 1 : size(); }
84 
85  /// @returns The product of the literal probabilities.
86  ///
87  /// @pre Events are initialized with expressions.
88  double p() const;
89 
90  /// @returns A read proxy iterator that points to the first element.
91  auto begin() const {
92  return boost::make_transform_iterator(data_.begin(),
93  LiteralExtractor{graph_});
94  }
95 
96  /// @returns A sentinel iterator signifying the end of iteration.
97  auto end() const {
98  return boost::make_transform_iterator(data_.end(),
99  LiteralExtractor{graph_});
100  }
101 
102  private:
103  const std::vector<int>& data_; ///< The collection of event indices.
104  const Pdag& graph_; ///< The host graph.
105 };

174 /// Fault tree analysis functionality.
175 /// The analysis must be done on
176 /// a validated and fully initialized fault trees.
177 /// After initialization of the analysis,
178 /// the fault tree under analysis should not change;
179 /// otherwise, the success of the analysis is not guaranteed,
180 /// and the results may be invalid.
181 /// After the requested analysis is done,
182 /// the fault tree can be changed without restrictions.
183 /// However, other analyses may rely on unchanged fault tree
184 /// to use the results of this fault tree analysis.
185 ///
186 /// To conduct a new analysis on the changed fault tree,
187 /// a new FaultTreeAnalysis object must be created.
188 /// In general, rerunning the same analysis twice
189 /// will mess up the analysis and corrupt the previous results.
190 ///
191 /// @warning Run analysis only once.
192 /// One analysis per FaultTreeAnalysis object.
193 class FaultTreeAnalysis : public Analysis {
194  public:
195  /// Traverses a valid fault tree from the root gate
196  /// to collect databases of events, gates,
197  /// and other members of the fault tree.
198  /// The passed fault tree must be pre-validated without cycles,
199  /// and its events must be fully initialized.
200  ///
201  /// @param[in] root The top event of the fault tree to analyze.
202  /// @param[in] settings Analysis settings for all calculations.
203  /// @param[in] model The Model containing substitutions if any.
204  ///
205  /// @note It is assumed that analysis is done only once.
206  ///
207  /// @warning If the fault tree structure is changed,
208  /// this analysis does not incorporate the changed structure.
209  /// Moreover, the analysis results may get corrupted.
210  FaultTreeAnalysis(const mef::Gate& root, const Settings& settings,
211  const mef::Model* model = nullptr);
212 
213  virtual ~FaultTreeAnalysis() = default;
214 
215  /// @returns The top gate that is passed to the analysis.
216  const mef::Gate& top_event() const { return top_event_; }
217 
218  /// Analyzes the fault tree and performs computations.
219  /// This function must be called
220  /// only after initializing the fault tree
221  /// with or without its probabilities.
222  ///
223  /// @note This function is expected to be called only once.
224  ///
225  /// @warning If the original fault tree is invalid,
226  /// this function will not throw or indicate any errors.
227  /// The behavior is undefined for invalid fault trees.
228  /// @warning If the fault tree structure has changed
229  /// since the construction of the analysis,
230  /// the analysis will be invalid or fail.
231  void Analyze() noexcept;
232 
233  /// @returns A collection of Boolean products as the analysis results.
234  ///
235  /// @pre The analysis is done.
236  const ProductContainer& products() const {
237  assert(products_ && "The analysis is not done!");
238  return *products_;
239  }
240 
241  protected:
242  /// @returns Pointer to the PDAG representing the fault tree.
243  const Pdag* graph() const { return graph_.get(); }
244 
245  private:
246  /// Preprocesses a PDAG for future analysis with a specific algorithm.
247  ///
248  /// @param[in,out] graph A valid PDAG for analysis.
249  ///
250  /// @post The graph transformation is semantically equivalent/isomorphic.
251  virtual void Preprocess(Pdag* graph) noexcept = 0;
252 
253  /// Generates a sum of products from a preprocessed PDAG.
254  ///
255  /// @param[in] graph The analysis PDAG.
256  ///
257  /// @returns The set of products.
258  ///
259  /// @pre The graph is specifically preprocessed for the algorithm.
260  ///
261  /// @post The result ZBDD lives as long as the host analysis.
262  virtual const Zbdd& GenerateProducts(const Pdag* graph) noexcept = 0;
263 
264  /// Stores resultant sets of products for future reporting.
265  ///
266  /// @param[in] products Sets with indices of events from calculations.
267  /// @param[in] graph PDAG with basic event indices and pointers.
268  void Store(const Zbdd& products, const Pdag& graph) noexcept;
269 
270  const mef::Gate& top_event_; ///< The root of the graph under analysis.
271  const mef::Model* model_; ///< The optional Model with substitutions.
272  std::unique_ptr<Pdag> graph_; ///< PDAG of the fault tree.
273  std::unique_ptr<const ProductContainer> products_; ///< Container of results.
274 };
