81 /// @throws ValidityError Invalid attribute setting.
82 void AttachLabelAndAttributes(const xml::Element& xml_element,
83  Element* element) {
84  if (std::optional<xml::Element> label = xml_element.child("label")) {
85  assert(element->label().empty() && "Resetting element label.");
86  element->label(std::string(label->text()));

97  std::string(attribute.attribute("type"))});

99  err << boost::errinfo_at_line(attribute.line());
100  throw;
101  }
102  }

226  CLOCK(setup_time);
227  LOG(DEBUG1) << "Setting up for the analysis";

351 void Initializer::ProcessInputFile(const xml::Document& document) {
352  xml::Element root = document.root();
353  assert(root.name() == "opsa-mef");
354 
355  if (!model_) { // Create only one model for multiple files.
356  model_ = ConstructElement<Model>(root);
357  model_->mission_time().value(settings_.mission_time());
358  }
359 
360  for (const xml::Element& node : root.children()) {
361  if (node.name() == "define-initiating-event") {
362  std::unique_ptr<InitiatingEvent> initiating_event =
363  ConstructElement<InitiatingEvent>(node);
364  auto* ref_ptr = initiating_event.get();
365  Register(std::move(initiating_event), node);
366  tbd_.emplace_back(ref_ptr, node);
367 
368  } else if (node.name() == "define-rule") {
369  std::unique_ptr<Rule> rule = ConstructElement<Rule>(node);
370  auto* ref_ptr = rule.get();
371  Register(std::move(rule), node);
372  tbd_.emplace_back(ref_ptr, node);
373 
374  } else if (node.name() == "define-event-tree") {
375  DefineEventTree(node);
376 
377  } else if (node.name() == "define-fault-tree") {
378  DefineFaultTree(node);
379 
380  } else if (node.name() == "define-CCF-group") {
381  Register<CcfGroup>(node, "", RoleSpecifier::kPublic);
382 
383  } else if (node.name() == "define-alignment") {
384  std::unique_ptr<Alignment> alignment = ConstructElement<Alignment>(node);
385  auto* address = alignment.get();
386  Register(std::move(alignment), node);
387  tbd_.emplace_back(address, node);
388 
389  } else if (node.name() == "define-substitution") {
390  std::unique_ptr<Substitution> substitution =
391  ConstructElement<Substitution>(node);
392  auto* address = substitution.get();
393  Register(std::move(substitution), node);
394  tbd_.emplace_back(address, node);
395 
396  } else if (node.name() == "model-data") {
397  ProcessModelData(node);
398 
399  } else if (node.name() == "define-extern-library") {
400  if (!allow_extern_) {
401  SCRAM_THROW(
402  IllegalOperation("Loading external libraries is disallowed!"))
403  << boost::errinfo_file_name(node.filename())
404  << boost::errinfo_at_line(node.line());
405  }
406  DefineExternLibraries(node);
407  }
408  }

864  auto* ret_ptr = instruction.get();

923 template <class T, int N>
924 struct Initializer::Extractor {
925  /// Extracts expressions
926  /// to be passed to the constructor of expression T.
927  ///
928  /// @param[in] args XML elements containing the arguments.
929  /// @param[in] base_path Series of ancestor containers in the path with dots.
930  /// @param[in,out] init The host Initializer.
931  ///
932  /// @returns The extracted expression.
933  ///
934  /// @pre The XML args container size equals N.
935  std::unique_ptr<T> operator()(const xml::Element::Range& args,
936  const std::string& base_path,
937  Initializer* init) {
938  static_assert(N > 0, "The number of arguments can't be fewer than 1.");
939  return (*this)(args.begin(), args.end(), base_path, init);
940  }
941 
942  /// Extracts and accumulates expressions
943  /// to be passed to the constructor of expression T.
944  ///
945  /// @tparam Ts Expression types.
946  ///
947  /// @param[in] it The iterator in the argument container.
948  /// @param[in] it_end The end sentinel iterator of the argument container.
949  /// @param[in] base_path Series of ancestor containers in the path with dots.
950  /// @param[in,out] init The host Initializer.
951  /// @param[in] expressions Accumulated argument expressions.
952  ///
953  /// @returns The extracted expression.
954  ///
955  /// @pre The XML container has enough arguments.
956  template <class... Ts>
957  std::unique_ptr<T> operator()(xml::Element::Range::iterator it,
958  xml::Element::Range::iterator it_end,
959  const std::string& base_path, Initializer* init,
960  Ts&&... expressions) {
961  static_assert(N >= 0);
962 
963  if constexpr (N == 0) {
964  static_assert(sizeof...(Ts), "Unintended use case.");
965  assert(it == it_end && "Too many arguments in the args container.");
966  return std::make_unique<T>(std::forward<Ts>(expressions)...);
967 
968  } else {
969  assert(it != it_end && "Not enough arguments in the args container.");
970  return Extractor<T, N - 1>()(std::next(it), it_end, base_path, init,
971  std::forward<Ts>(expressions)...,
972  init->GetExpression(*it, base_path));
973  }
974  }
975 };

1158  auto* ret_ptr = expression.get();
1159  model_->Add(std::move(expression));
1160  return ret_ptr;
1161  };
1162  if (expr_type == "int") {
1163  int val = *expr_element.attribute<int>("value");
1164  return register_expression(std::make_unique<ConstantExpression>(val));
1165  }
1166  if (expr_type == "float") {
1167  double val = *expr_element.attribute<double>("value");
1168  return register_expression(std::make_unique<ConstantExpression>(val));
1169  }
1170  if (expr_type == "bool") {
1171  bool val = *expr_element.attribute<bool>("value");
1172  return val ? &ConstantExpression::kOne : &ConstantExpression::kZero;
1173  }
1174  if (expr_type == "pi")
1175  return &ConstantExpression::kPi;
1176 
1177  if (expr_type == "test-initiating-event") {
1178  return register_expression(std::make_unique<TestInitiatingEvent>(
1179  std::string(expr_element.attribute("name")), model_->context()));
1180  }
1181  if (expr_type == "test-functional-event") {
1182  return register_expression(std::make_unique<TestFunctionalEvent>(
1183  std::string(expr_element.attribute("name")),
1184  std::string(expr_element.attribute("state")), model_->context()));
1185  }
1186

1190  auto& ret =
1191  model_->Get<ExternFunction<void>>(expr_element.attribute("name"));
1192  ret.usage(true);
1193  return &ret;
1194

1196  err << boost::errinfo_at_line(expr_element.line());
1197  throw;
1198  }
1199  }();

1202  for (const xml::Element& node : expr_element.children())
1203  expr_args.push_back(GetExpression(node, base_path));
1204

1206  return register_expression(extern_function->apply(std::move(expr_args)));
1207  } catch (ValidityError& err) {
1208  err << boost::errinfo_at_line(expr_element.line());
1209  throw;
1210  }
1211  }

1216  try {

1218  expr_type)(expr_element.children(), base_path, this));
1219  // Register for late validation after ensuring no cycles.

1223  err << boost::errinfo_at_line(expr_element.line());
1224  throw;

1407 /// All the allowed extern function parameter types.
1408 ///
1409 /// @note Template code may be optimized for these types only.
1410 enum class ExternParamType { kInt = 1, kDouble };
1411 const int kExternTypeBase = 3; ///< The information base for encoding.
1412 const int kMaxNumParam = 5; ///< The max number of args (excludes the return).
1413 const int kNumInterfaces = 126; ///< All possible interfaces.
1414 
1415 /// Encodes parameter types kExternTypeBase base number.
1416 ///
1417 /// @tparam SinglePassRange The forward range type
1418 ///
1419 /// @param[in] args The non-empty XML elements encoding parameter types.
1420 ///
1421 /// @returns Unique integer encoding parameter types.
1422 ///
1423 /// @pre The number of parameters is less than log_base(max int).
1424 template <class SinglePassRange>
1425 int Encode(const SinglePassRange& args) noexcept {
1426  assert(!args.empty());
1427  auto to_digit = [](const xml::Element& node) -> int {
1428  std::string_view name = node.name();
1429  return static_cast<int>([&name] {
1430  if (name == "int")
1431  return ExternParamType::kInt;
1432  assert(name == "double");
1433  return ExternParamType::kDouble;
1434  }());
1435  };
1436 
1437  int ret = 0;
1438  int base_power = 1; // Base ^ (pos - 1).
1439  for (const xml::Element& node : args) {
1440  ret += base_power * to_digit(node);
1441  base_power *= kExternTypeBase;
1442  }
1443 
1444  return ret;
1445 }
1446 
1447 /// Encodes function parameter types at compile-time.
1448 template <typename T, typename... Ts>
1449 constexpr int Encode(int base_power = 1) noexcept {
1450  if constexpr (sizeof...(Ts)) {
1451  return Encode<T>(base_power) + Encode<Ts...>(base_power * kExternTypeBase);
1452 
1453  } else if constexpr (std::is_same_v<T, int>) {
1454  return base_power * static_cast<int>(ExternParamType::kInt);
1455 
1456  } else {
1457  static_assert(std::is_same_v<T, double>);
1458  return base_power * static_cast<int>(ExternParamType::kDouble);
1459  }
1460 }
