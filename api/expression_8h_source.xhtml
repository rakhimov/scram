59 /// Abstract base class for all sorts of expressions to describe events.
60 /// This class also acts like a connector for parameter nodes
61 /// and may create cycles.
62 /// Expressions are not expected to be shared
63 /// except for parameters.
64 /// In addition, expressions are not expected to be changed
65 /// after validation phases.
66 class Expression : private boost::noncopyable {
67  public:
68  /// Constructor for use by derived classes
69  /// to register their arguments.
70  ///
71  /// @param[in] args Arguments of this expression.
72  explicit Expression(std::vector<Expression*> args = {});
73 
74  virtual ~Expression() = default;
75 
76  /// @returns A set of arguments of the expression.
77  const std::vector<Expression*>& args() const { return args_; }
78 
79  /// Validates the expression.
80  /// This late validation is due to parameters that are defined late.
81  ///
82  /// @throws DomainError The argument value domains are invalid.
83  /// @throws ValidityError The arguments are invalid for setup.
84  virtual void Validate() const {}
85 
86  /// @returns The mean value of this expression.
87  virtual double value() noexcept = 0;
88 
89  /// @returns The domain interval for validation purposes only.
90  virtual Interval interval() noexcept {
91  double value = this->value();
92  return Interval::closed(value, value);
93  }
94 
95  /// Determines if the value of the expression contains deviate expressions.
96  /// The default logic is to check arguments with uncertainties for sampling.
97  /// Derived expression classes must decide
98  /// if they don't have arguments,
99  /// or if they are random deviates.
100  ///
101  /// @returns true if the expression's value deviates from its mean.
102  /// @returns false if the expression's value does not need sampling.
103  ///
104  /// @warning Improper registration of arguments
105  /// may yield silent failure.
106  virtual bool IsDeviate() noexcept;
107 
108  /// @returns A sampled value of this expression.
109  double Sample() noexcept;
110 
111  /// This routine resets the sampling to get new values.
112  /// All the arguments are called to reset themselves.
113  /// If this expression was not sampled,
114  /// its arguments are not going to get any calls.
115  void Reset() noexcept;
116 
117  protected:
118  /// Registers an additional argument expression.
119  ///
120  /// @param[in] arg An argument expression used by this expression.
121  void AddArg(Expression* arg) { args_.push_back(arg); }
122 
123  private:
124  /// Runs sampling of the expression.
125  /// Derived concrete classes must provide the calculation.
126  ///
127  /// @returns A sampled value of this expression.
128  virtual double DoSample() noexcept = 0;
129 
130  std::vector<Expression*> args_; ///< Expression's arguments.
131  double sampled_value_; ///< The sampled value.
132  bool sampled_; ///< Indication if the expression is already sampled.
133 };

139 class ExpressionFormula : public Expression {

164 /// Unary expression.
165 template <typename T>
166 class NaryExpression<T, 1> : public ExpressionFormula<NaryExpression<T, 1>> {
167  public:
168  /// @param[in] expression The single argument.
169  explicit NaryExpression(Expression* expression)
170  : ExpressionFormula<NaryExpression<T, 1>>({expression}),
171  expression_(*expression) {}
172 
173  void Validate() const override {}
174 
175  Interval interval() noexcept override {
176  Interval arg_interval = expression_.interval();
177  double max_value = T()(arg_interval.upper());
178  double min_value = T()(arg_interval.lower());
179  auto min_max = std::minmax(max_value, min_value);
180  return Interval::closed(min_max.first, min_max.second);
181  }

206  double max_max = T()(interval_one.upper(), interval_two.upper());

233 /// Multivariate expression.
234 template <typename T>
235 class NaryExpression<T, -1> : public ExpressionFormula<NaryExpression<T, -1>> {
236  public:
237  /// Checks the number of provided arguments upon initialization.
238  ///
239  /// @param[in] args Arguments of this expression.
240  ///
241  /// @throws ValidityError The number of arguments is fewer than 2.
242  explicit NaryExpression(std::vector<Expression*> args)
243  : ExpressionFormula<NaryExpression<T, -1>>(std::move(args)) {
244  detail::EnsureMultivariateArgs(Expression::args());
245  }
246 
247  void Validate() const override {}
248 
249  Interval interval() noexcept override {
250  auto it = Expression::args().begin();
251  Interval first_arg_interval = (*it)->interval();
252  double max_value = first_arg_interval.upper();
253  double min_value = first_arg_interval.lower();
254  for (++it; it != Expression::args().end(); ++it) {
255  Interval next_arg_interval = (*it)->interval();
256  double arg_max = next_arg_interval.upper();
257  double arg_min = next_arg_interval.lower();
258  double max_max = T()(max_value, arg_max);
259  double max_min = T()(max_value, arg_min);
260  double min_max = T()(min_value, arg_max);
261  double min_min = T()(min_value, arg_min);
262  std::tie(min_value, max_value) =
263  std::minmax({max_max, max_min, min_max, min_min});
264  }
265  assert(min_value <= max_value);
266  return Interval::closed(min_value, max_value);
267  }

269  /// Computes the expression value with a given argument value extractor.
270  template <typename F>
271  double Compute(F&& eval) noexcept {
272  auto it = Expression::args().begin();
273  double result = eval(*it);
274  for (++it; it != Expression::args().end(); ++it) {
275  result = T()(result, eval(*it));
276  }
277  return result;
278  }
