38 void Substitution::Validate() const {
39  assert(hypothesis_ && "Missing substitution hypothesis.");
40  if (ext::any_of(hypothesis_->args(), [](const Formula::Arg& arg) {
41  return !std::holds_alternative<BasicEvent*>(arg.event);
42  })) {
43  SCRAM_THROW(ValidityError(
44  "Substitution hypothesis must be built over basic events only."))
45  << errinfo_element(Element::name(), kTypeString);
46  }
47 
48  if (ext::any_of(hypothesis_->args(),
49  [](const Formula::Arg& arg) { return arg.complement; })) {
50  SCRAM_THROW(ValidityError("Substitution hypotheses must be coherent."))
51  << errinfo_element(Element::name(), kTypeString);
52  }
53 
54  if (declarative()) {
55  switch (hypothesis_->connective()) {
56  case kNull:
57  case kAnd:
58  case kAtleast:
59  case kOr:
60  break;
61  default:
62  SCRAM_THROW(ValidityError("Substitution hypotheses must be coherent."))
63  << errinfo_element(Element::name(), kTypeString)
64  << errinfo_connective(
65  kConnectiveToString[hypothesis_->connective()]);
66  }
67  const bool* constant = std::get_if<bool>(&target_);
68  if (constant && *constant)
69  SCRAM_THROW(ValidityError("Substitution has no effect."))
70  << errinfo_element(Element::name(), kTypeString);
71  } else { // Non-declarative.
72  switch (hypothesis_->connective()) {
73  case kNull:
74  case kAnd:
75  case kOr:
76  break;
77  default:
78  SCRAM_THROW(
79  ValidityError("Non-declarative substitution hypotheses only allow "
80  "AND/OR/NULL connectives."))
81  << errinfo_element(Element::name(), kTypeString)
82  << errinfo_connective(
83  kConnectiveToString[hypothesis_->connective()]);
84  }
85  const bool* constant = std::get_if<bool>(&target_);
86  if (constant && !*constant)
87  SCRAM_THROW(ValidityError("Substitution source set is irrelevant."))
88  << errinfo_element(Element::name(), kTypeString);
89  }
90 }

92 std::optional<Substitution::Type> Substitution::type() const {

95  [source_arg](const Formula::Arg& arg) {

100  auto is_mutually_exclusive = [](const Formula& formula) {
101  switch (formula.connective()) {
102  case kAtleast:
103  return formula.min_number() == 2;
104  case kAnd:
105  return formula.args().size() == 2;
106  default:
107  return false;
108  }
109  };
110 
111  if (source_.empty()) {
112  if (const bool* constant = std::get_if<bool>(&target_)) {
113  assert(!*constant && "Substitution has no effect.");
114  if (is_mutually_exclusive(*hypothesis_))
115  return kDeleteTerms;
116  } else if (std::holds_alternative<BasicEvent*>(target_)) {
117  if (hypothesis_->connective() == kAnd)
118  return kRecoveryRule;
119  }
120  return {};
121  }
122  if (!std::holds_alternative<BasicEvent*>(target_))
123  return {};
124  if (hypothesis_->connective() != kAnd && hypothesis_->connective() != kNull)
125  return {};
126 
127  if (source_.size() == hypothesis_->args().size()) {
128  if (ext::all_of(source_, in_hypothesis))
129  return kRecoveryRule;
130  } else if (source_.size() == 1) {
131  if (in_hypothesis(source_.front()))
132  return kExchangeEvent;
133  }
134  return {};
135 }
