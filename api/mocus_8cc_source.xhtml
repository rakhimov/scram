39 void Mocus::Analyze(const Pdag*) noexcept {

52 std::unique_ptr<zbdd::CutSetContainer>
53 Mocus::AnalyzeModule(const Gate& gate, const Settings& settings) noexcept {
54  assert(gate.module() && "Expected only module gates.");
55  CLOCK(gen_time);
56  LOG(DEBUG3) << "Finding cut sets from module: G" << gate.index();
57  LOG(DEBUG4) << "Limit on product order: " << settings.limit_order();
58  std::unordered_map<int, const Gate*> gates;
59  auto add_gates = [&gates](const auto& args) {
60  for (const Gate::ConstArg<Gate>& arg : args)
61  gates.emplace(arg.first, &arg.second);
62  };
63  add_gates(gate.args<Gate>());
64  const int kMaxVariableIndex =
65  Pdag::kVariableStartIndex + graph_->basic_events().size() - 1;
66  auto container = std::make_unique<zbdd::CutSetContainer>(
67  kSettings_, gate.index(), kMaxVariableIndex);
68  container->Merge(container->ConvertGate(gate));
69  while (int next_gate_index = container->GetNextGate()) {
70  LOG(DEBUG5) << "Expanding gate G" << next_gate_index;
71  const Gate* next_gate = gates.find(next_gate_index)->second;
72  add_gates(next_gate->args<Gate>());
73 
74  container->Merge(container->ExpandGate(
75  container->ConvertGate(*next_gate),
76  container->ExtractIntermediateCutSets(next_gate_index)));
77  }
78  container->Minimize();
79  container->Log();
80  LOG(DEBUG3) << "G" << gate.index()
81  << " cut set generation time: " << DUR(gen_time);
82  if (!gate.coherent()) {
83  container->EliminateComplements();
84  container->Minimize();
85  }
86  for (const auto& entry : container->GatherModules()) {
87  int index = entry.first;
88  assert(index > 0 && "No complement modules are expected.");
89  int limit = entry.second.second;
90  assert(limit >= 0 && "Order cut-off is not strict.");
91  bool coherent = entry.second.first;
92  if (limit == 0 && coherent) { // Unity is impossible.
93  auto empty_zbdd = std::make_unique<zbdd::CutSetContainer>(
94  kSettings_, index, kMaxVariableIndex);
95  container->JoinModule(index, std::move(empty_zbdd));
96  continue;
97  }
98  Settings adjusted(settings);
99  adjusted.limit_order(limit);
100  container->JoinModule(index,
101  AnalyzeModule(*gates.find(index)->second, adjusted));
102  }
103  container->EliminateConstantModules();
104  container->Minimize();
105  return container;
106 }
