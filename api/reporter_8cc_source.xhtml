71 void Reporter::Report(const core::RiskAnalysis& risk_an, std::FILE* out,
72  bool indent) {
73  xml::Stream xml_stream(out, indent);
74  xml::StreamElement report = xml_stream.root("report");
75  ReportInformation(risk_an, &report);
76 
77  if (risk_an.results().empty() && risk_an.event_tree_results().empty())
78  return;
79  TIMER(DEBUG1, "Reporting analysis results");
80  xml::StreamElement results = report.AddChild("results");
81  if (risk_an.settings().probability_analysis()) {
82  for (const core::RiskAnalysis::EtaResult& result :
83  risk_an.event_tree_results()) {
84  ReportResults(result, &results);
85  }
86  }
87 
88  for (const core::RiskAnalysis::Result& result : risk_an.results()) {
89  if (result.fault_tree_analysis)
90  ReportResults(result.id, *result.fault_tree_analysis,
91  result.probability_analysis.get(), &results);
92 
93  if (result.probability_analysis)
94  ReportResults(result.id, *result.probability_analysis, &results);
95 
96  if (result.importance_analysis)
97  ReportResults(result.id, *result.importance_analysis, &results);
98 
99  if (result.uncertainty_analysis)
100  ReportResults(result.id, *result.uncertainty_analysis, &results);
101  }
102 }

104 void Reporter::Report(const core::RiskAnalysis& risk_an,
105  const std::string& file, bool indent) {
106  std::unique_ptr<std::FILE, decltype(&std::fclose)> fp(
107  std::fopen(file.c_str(), "w"), &std::fclose);

109  if (!fp) {

116  throw;
117  }

375 void Reporter::ReportResults(const core::RiskAnalysis::Result::Id& id,
376  const core::FaultTreeAnalysis& fta,
377  const core::ProbabilityAnalysis* prob_analysis,
378  xml::StreamElement* results) {
379  TIMER(DEBUG2, "Reporting products");
380  xml::StreamElement sum_of_products = results->AddChild("sum-of-products");
381  scram::PutId(id, &sum_of_products);
382 
383  std::string warning = fta.warnings();
384  if (prob_analysis && prob_analysis->warnings().empty() == false)
385  warning += (warning.empty() ? "" : "; ") + prob_analysis->warnings();
386  if (!warning.empty())
387  sum_of_products.SetAttribute("warning", warning);
388 
389  sum_of_products
390  .SetAttribute("basic-events", fta.products().product_events().size())
391  .SetAttribute("products", fta.products().size());
392 
393  if (prob_analysis)
394  sum_of_products.SetAttribute("probability", prob_analysis->p_total());
395 
396  if (fta.products().empty() == false) {
397  sum_of_products.SetAttribute(
398  "distribution",
399  boost::join(fta.products().distribution() |
400  boost::adaptors::transformed(
401  [](int number) { return std::to_string(number); }),
402  " "));
403  }
404 
405  double sum = 0; // Sum of probabilities for contribution calculations.
406  if (prob_analysis) {
407  for (const core::Product& product_set : fta.products())
408  sum += product_set.p();
409  }
410  for (const core::Product& product_set : fta.products()) {
411  xml::StreamElement product = sum_of_products.AddChild("product");
412  product.SetAttribute("order", product_set.order());
413  if (prob_analysis) {
414  double prob = product_set.p();
415  product.SetAttribute("probability", prob);
416  if (sum != 0)
417  product.SetAttribute("contribution", prob / sum);
418  }
419  for (const core::Literal& literal : product_set) {
420  ReportLiteral(literal, &product);
421  }
422  }
423 }

495 void Reporter::ReportResults(const core::RiskAnalysis::Result::Id& id,
496  const core::UncertaintyAnalysis& uncert_analysis,
497  xml::StreamElement* results) {
498  xml::StreamElement measure = results->AddChild("measure");
499  scram::PutId(id, &measure);
500  if (!uncert_analysis.warnings().empty()) {
501  measure.SetAttribute("warning", uncert_analysis.warnings());
502  }
503  measure.AddChild("mean").SetAttribute("value", uncert_analysis.mean());
504  measure.AddChild("standard-deviation")
505  .SetAttribute("value", uncert_analysis.sigma());
506 
507  measure.AddChild("confidence-range")
508  .SetAttribute("percentage", "95")
509  .SetAttribute("lower-bound", uncert_analysis.confidence_interval().first)
510  .SetAttribute("upper-bound",
511  uncert_analysis.confidence_interval().second);
512 
513  measure.AddChild("error-factor")
514  .SetAttribute("percentage", "95")
515  .SetAttribute("value", uncert_analysis.error_factor());
516  {
517  xml::StreamElement quantiles = measure.AddChild("quantiles");
518  int num_quantiles = uncert_analysis.quantiles().size();
519  quantiles.SetAttribute("number", num_quantiles);
520  double prev_bound = 0;
521  double delta = 1.0 / num_quantiles;
522  for (int i = 0; i < num_quantiles; ++i) {
523  double upper = uncert_analysis.quantiles()[i];
524  double value = delta * (i + 1);
525  quantiles.AddChild("quantile")
526  .SetAttribute("number", i + 1)
527  .SetAttribute("value", value)
528  .SetAttribute("lower-bound", prev_bound)
529  .SetAttribute("upper-bound", upper);
530  prev_bound = upper;
531  }
532  }
533  {
534  xml::StreamElement hist = measure.AddChild("histogram");
535  int num_bins = uncert_analysis.distribution().size() - 1;
536  hist.SetAttribute("number", num_bins);
537  for (int i = 0; i < num_bins; ++i) {
538  double lower = uncert_analysis.distribution()[i].first;
539  double upper = uncert_analysis.distribution()[i + 1].first;
540  double value = uncert_analysis.distribution()[i].second;
541  hist.AddChild("bin")
542  .SetAttribute("number", i + 1)
543  .SetAttribute("value", value)
544  .SetAttribute("lower-bound", lower)
545  .SetAttribute("upper-bound", upper);
546  }
547  }

561 template <class T>
562 void Reporter::ReportBasicEvent(const mef::BasicEvent& basic_event,
563  xml::StreamElement* parent, const T& add_data) {
564  const auto* ccf_event = dynamic_cast<const mef::CcfEvent*>(&basic_event);
565  if (!ccf_event) {

568  add_data(&element);

571  xml::StreamElement element = parent->AddChild("ccf-event");
572  element.SetAttribute("ccf-group", ccf_group.id())
573  .SetAttribute("order", ccf_event->members().size())
574  .SetAttribute("group-size", ccf_group.members().size());
575  add_data(&element);
576  for (const mef::Gate* member : ccf_event->members()) {
577  element.AddChild("basic-event").SetAttribute("name", member->name());
