34 void Print(const ProductContainer& products) {
35  if (products.empty()) {
36  std::cerr << "No products!" << std::endl;
37  return;
38  }
39  if (products.begin()->empty()) {
40  assert(products.size() == 1 && "Unity case must have only one product.");
41  std::cerr << "Single Unity product." << std::endl;
42  return;
43  }
44  std::cerr << " " << products.size() << " : {";
45  for (int i : products.distribution())
46  std::cerr << " " << i;
47  std::cerr << " }\n\n";

49  using LiteralPtr = std::pair<bool, const mef::BasicEvent*>;
50  struct Comparator {
51  bool operator()(const LiteralPtr& lhs, const LiteralPtr& rhs) const {
52  return lhs.second->id() < rhs.second->id();
53  }
54  };
55 
56  using ProductSet = boost::container::flat_set<LiteralPtr, Comparator>;
57  std::vector<ProductSet> to_print;
58  to_print.reserve(products.size());
59  for (const Product& product : products) {
60  ProductSet ids;
61  ids.reserve(product.size());
62  for (const Literal& literal : product) {
63  ids.emplace(literal.complement, &literal.event);
64  }
65  to_print.push_back(std::move(ids));
66  }
67  boost::sort(to_print, [](const ProductSet& lhs, const ProductSet& rhs) {
68  if (lhs.size() == rhs.size())
69  return lhs < rhs;
70  return lhs.size() < rhs.size();
71  });
72  assert(!to_print.front().empty() && "Failure of the analysis with Unity!");
73 
74  for (const auto& product : to_print) {
75  for (const auto& [complement, event] : product) {
76  std::cerr << " ";
77  if (complement)
78  std::cerr << "~";
79  std::cerr << event->id();
80  }
81  std::cerr << "\n";
82  }
83  std::cerr << std::endl;
84 }

86 ProductContainer::ProductContainer(const Zbdd& products,
87  const Pdag& graph) noexcept
88  : products_(products), graph_(graph), size_(0) {
89  Pdag::IndexMap<bool> filter(graph_.basic_events().size());
90  for (const std::vector<int>& product : products_) {
91  int order_index = product.empty() ? 0 : product.size() - 1;
92  if (distribution_.size() <= order_index)
93  distribution_.resize(order_index + 1);
94  distribution_[order_index]++;
95  ++size_;
96 
97  for (int i : product) {
98  i = std::abs(i);
99  if (filter[i])
100  continue;
101  filter[i] = true;
102  product_events_.insert(graph_.basic_events()[i]);
103  }
104  }
105 }

120 void FaultTreeAnalysis::Analyze() noexcept {
121  CLOCK(analysis_time);
122  graph_ = std::make_unique<Pdag>(top_event_,
123  Analysis::settings().ccf_analysis(), model_);
124  this->Preprocess(graph_.get());
125 #ifndef NDEBUG
126  if (Analysis::settings().preprocessor)
127  return; // Preprocessor only option.
128 #endif
129  CLOCK(algo_time);
130  LOG(DEBUG2) << "Launching the algorithm...";
131  const Zbdd& products = this->GenerateProducts(graph_.get());
132  LOG(DEBUG2) << "The algorithm finished in " << DUR(algo_time);
133  LOG(DEBUG2) << "# of products: " << products.size();
134 
135  Analysis::AddAnalysisTime(DUR(analysis_time));
136  CLOCK(store_time);

143  // Special cases of sets.

146  } else if (products.base()) {

154  "Miscalculated product sets with larger-than-required order.");
