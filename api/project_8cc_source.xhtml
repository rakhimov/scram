48 std::string normalize(const std::string& file_path, const fs::path& base_path) {
49  fs::path abs_path = fs::absolute(file_path, base_path).generic_string();

61 Project::Project(const std::string& project_file) {
62  static xml::Validator validator(env::project_schema());
63 
64  if (fs::exists(project_file) == false) {
65  SCRAM_THROW(IOError("The configuration file does not exist."))
66  << boost::errinfo_file_name(project_file);
67  }
68 
69  xml::Document document(project_file);
70  xml::Element root = document.root();
71  std::string_view version = root.attribute("version");
72  if (root.name() == "scram" && !version.empty()) {
73  try {
74  std::optional<std::array<int, 3>> numbers = ext::extract_version(version);
75  if (!numbers)
76  SCRAM_THROW(xml::ValidityError("Invalid version string"));
77  std::array<int, 3> current_numbers = {
78  SCRAM_VERSION_MAJOR, SCRAM_VERSION_MINOR, SCRAM_VERSION_MICRO};
79  if (numbers > current_numbers)
80  SCRAM_THROW(VersionError("Version incompatibility"));
81 
82  } catch (Error& err) {
83  err << errinfo_value(std::string(version))
84  << xml::errinfo_element("scram") << xml::errinfo_attribute("version")
85  << boost::errinfo_at_line(root.line())
86  << boost::errinfo_file_name(root.filename());
87  throw;
88  }
89  }
90  validator.validate(document);
91  assert(root.name() == "scram");
92  fs::path base_path = fs::path(project_file).parent_path();
93  GatherInputFiles(root, base_path);
94 
95  try {
96  GatherOptions(root);
97  } catch (Error& err) {
98  err << boost::errinfo_file_name(project_file);
99  throw;
100  }
101 }

103 void Project::GatherInputFiles(const xml::Element& root,
104  const fs::path& base_path) {
105  std::optional<xml::Element> input_files = root.child("model");
106  if (!input_files)
107  return;
108  for (xml::Element input_file : input_files->children()) {
109  assert(input_file.name() == "file");
110  input_files_.push_back(
111  normalize(std::string(input_file.text()), base_path));
112  }

115 void Project::GatherOptions(const xml::Element& root) {
116  std::optional<xml::Element> options_element = root.child("options");
117  if (!options_element)
118  return;
119  // The loop is used instead of query
120  // because the order of options matters,
121  // yet this function should not know what the order is.
122  for (xml::Element option_group : options_element->children()) {
123  try {
124  std::string_view name = option_group.name();
125  if (name == "algorithm") {
126  settings_.algorithm(option_group.attribute("name"));
127 
128  } else if (name == "prime-implicants") {
129  settings_.prime_implicants(true);
130 
131  } else if (name == "approximation") {
132  settings_.approximation(option_group.attribute("name"));
133 
134  } else if (name == "limits") {
135  SetLimits(option_group);
136  }
137  } catch (SettingsError& err) {
138  err << boost::errinfo_at_line(option_group.line());
139  throw;
140  }
141  }
142  if (std::optional<xml::Element> analysis_group =
143  options_element->child("analysis")) {
144  try {
145  SetAnalysis(*analysis_group);
146  } catch (SettingsError& err) {
147  err << boost::errinfo_at_line(analysis_group->line());
148  throw;
149  }
150  }
151 }

153 void Project::SetAnalysis(const xml::Element& analysis) {
154  auto set_flag = [&analysis](const char* tag, auto setter) {
155  if (std::optional<bool> flag = analysis.attribute<bool>(tag))
156  setter(*flag);
157  };
158  set_flag("probability",
159  [this](bool flag) { settings_.probability_analysis(flag); });
160  set_flag("importance",
161  [this](bool flag) { settings_.importance_analysis(flag); });
162  set_flag("uncertainty",
163  [this](bool flag) { settings_.uncertainty_analysis(flag); });
164  set_flag("ccf", [this](bool flag) { settings_.ccf_analysis(flag); });
165  set_flag("sil",
166  [this](bool flag) { settings_.safety_integrity_levels(flag); });
