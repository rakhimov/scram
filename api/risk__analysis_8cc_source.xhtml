36 void RiskAnalysis::Analyze() noexcept {
37  assert(results_.empty() && "Rerunning the analysis.");
38  // Set the seed for the pseudo-random number generator if given explicitly.
39  // Otherwise it defaults to the implementation dependent value.
40  if (Analysis::settings().seed() >= 0)
41  mef::RandomDeviate::seed(Analysis::settings().seed());
42 
43  if (model_->alignments().empty()) {
44  RunAnalysis();
45  } else {
46  for (const mef::Alignment& alignment : model_->alignments()) {
47  for (const mef::Phase& phase : alignment.phases())
48  RunAnalysis(Context{alignment, phase});
49  }
50  }
51 }

53 void RiskAnalysis::RunAnalysis(std::optional<Context> context) noexcept {
54  std::vector<std::pair<mef::HouseEvent*, bool>> house_events;
55  /// Restores the model after application of the context.
56  ext::scope_guard restorator(
57  [&house_events, this, init_time = model_->mission_time().value()] {
58  model_->mission_time().value(init_time);
59  Analysis::settings().mission_time(init_time);
60 
61  for (const std::pair<mef::HouseEvent*, bool>& entry : house_events)
62  entry.first->state(entry.second);
63  });
64 
65  if (context) {
66  double mission_time =
67  context->phase.time_fraction() * model_->mission_time().value();
68  model_->mission_time().value(mission_time);
69  Analysis::settings().mission_time(mission_time);
70 
71  for (const mef::SetHouseEvent* instruction :
72  context->phase.instructions()) {
73  auto it = model_->table<mef::HouseEvent>().find(instruction->name());
74  assert(it != model_->table<mef::HouseEvent>().end() &&
75  "Invalid instruction.");
76  mef::HouseEvent& house_event = *it;
77  if (house_event.state() != instruction->state()) {
78  house_events.emplace_back(&house_event, house_event.state());
79  house_event.state(instruction->state());
80  }
81  }
82  }
83 
84  for (const mef::InitiatingEvent& initiating_event :
85  model_->initiating_events()) {
86  if (initiating_event.event_tree()) {
87  LOG(INFO) << "Running event tree analysis: " << initiating_event.name();
88  auto eta = std::make_unique<EventTreeAnalysis>(
89  initiating_event, Analysis::settings(), model_->context());
90  eta->Analyze();
91  for (EventTreeAnalysis::Result& result : eta->sequences()) {
92  const mef::Sequence& sequence = result.sequence;
93  LOG(INFO) << "Running analysis for sequence: " << sequence.name();
94  results_.push_back(
95  {{std::pair<const mef::InitiatingEvent&, const mef::Sequence&>{
96  initiating_event, sequence},
97  context}});
98  RunAnalysis(*result.gate, &results_.back());
99  if (result.is_expression_only) {
100  results_.back().fault_tree_analysis = nullptr;
101  results_.back().importance_analysis = nullptr;
102  }
103  if (Analysis::settings().probability_analysis())
104  result.p_sequence = results_.back().probability_analysis->p_total();
105  LOG(INFO) << "Finished analysis for sequence: " << sequence.name();
106  }
107  event_tree_results_.push_back(
108  {initiating_event, context, std::move(eta)});
109  LOG(INFO) << "Finished event tree analysis: " << initiating_event.name();
110  }
111  }
112 
113  for (const mef::FaultTree& ft : model_->fault_trees()) {
114  for (const mef::Gate* target : ft.top_events()) {
115  LOG(INFO) << "Running analysis for gate: " << target->id();
116  results_.push_back({{target, context}});
117  RunAnalysis(*target, &results_.back());
118  LOG(INFO) << "Finished analysis for gate: " << target->id();
119  }
120  }
121 }

123 void RiskAnalysis::RunAnalysis(const mef::Gate& target,
124  Result* result) noexcept {
125  switch (Analysis::settings().algorithm()) {
126  case Algorithm::kBdd:
127  return RunAnalysis<Bdd>(target, result);
128  case Algorithm::kZbdd:
129  return RunAnalysis<Zbdd>(target, result);
130  case Algorithm::kMocus:
131  return RunAnalysis<Mocus>(target, result);
132  }

135 template <class Algorithm>
136 void RiskAnalysis::RunAnalysis(const mef::Gate& target,
137  Result* result) noexcept {
138  auto fta = std::make_unique<FaultTreeAnalyzer<Algorithm>>(
139  target, Analysis::settings(), model_);
140  fta->Analyze();
141  if (Analysis::settings().probability_analysis()) {
142  switch (Analysis::settings().approximation()) {
143  case Approximation::kNone:
144  RunAnalysis<Algorithm, Bdd>(fta.get(), result);
145  break;
146  case Approximation::kRareEvent:
147  RunAnalysis<Algorithm, RareEventCalculator>(fta.get(), result);
148  break;
149  case Approximation::kMcub:
150  RunAnalysis<Algorithm, McubCalculator>(fta.get(), result);
151  }
152  }
153  result->fault_tree_analysis = std::move(fta);
154 }

156 template <class Algorithm, class Calculator>
157 void RiskAnalysis::RunAnalysis(FaultTreeAnalyzer<Algorithm>* fta,
158  Result* result) noexcept {
159  auto pa = std::make_unique<ProbabilityAnalyzer<Calculator>>(
160  fta, &model_->mission_time());
161  pa->Analyze();
162  if (Analysis::settings().importance_analysis()) {
163  auto ia = std::make_unique<ImportanceAnalyzer<Calculator>>(pa.get());
164  ia->Analyze();
165  result->importance_analysis = std::move(ia);
166  }
167  if (Analysis::settings().uncertainty_analysis()) {
168  auto ua = std::make_unique<UncertaintyAnalyzer<Calculator>>(pa.get());
169  ua->Analyze();
170  result->uncertainty_analysis = std::move(ua);
171  }
172  result->probability_analysis = std::move(pa);
173 }
