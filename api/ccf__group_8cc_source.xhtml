82 void CcfGroup::AddFactor(Expression* factor, std::optional<int> level) {
83  int min_level = this->min_level();
84  if (!level)
85  level = prev_level_ ? (prev_level_ + 1) : min_level;
86 
87  if (*level <= 0 || members_.empty())
88  SCRAM_THROW(LogicError("Invalid CCF group factor setup."));
89 
90  if (*level < min_level) {
91  SCRAM_THROW(ValidityError("The CCF factor level (" +
92  std::to_string(*level) +
93  ") is less than the minimum level (" +
94  std::to_string(min_level) + ")."))
95  << errinfo_element(Element::name(), kTypeString);
96  }
97  if (members_.size() < *level) {
98  SCRAM_THROW(ValidityError("The CCF factor level " + std::to_string(*level) +
99  " is more than the number of members (" +
100  std::to_string(members_.size()) + ")"))
101  << errinfo_element(Element::name(), kTypeString);
102  }
103 
104  int index = *level - min_level;
105  if (index < factors_.size() && factors_[index].second != nullptr) {
106  SCRAM_THROW(ValidityError("Redefinition of CCF factor for level " +
107  std::to_string(*level)))
108  << errinfo_element(Element::name(), kTypeString);
109  }
110  if (index >= factors_.size())
111  factors_.resize(index + 1);
112 
113  factors_[index] = {*level, factor};
114  prev_level_ = *level;
115 }

139 void CcfGroup::ApplyModel() {
140  // Construct replacement proxy gates for member basic events.
141  std::vector<std::pair<Gate*, Formula::ArgSet>> proxy_gates;
142  for (BasicEvent* member : members_) {
143  auto new_gate = std::make_unique<Gate>(member->name(), member->base_path(),
144  member->role());
145  assert(member->id() == new_gate->id());
146  proxy_gates.push_back({new_gate.get(), {}});
147  member->ccf_gate(std::move(new_gate));
148  }
149 
150  ExpressionMap probabilities = this->CalculateProbabilities();
151  assert(probabilities.size() > 1);
152 
153  // Generate CCF events.
154  for (auto& [level, prob] : probabilities) {
155  using Iterator = decltype(proxy_gates)::iterator;
156  auto combination_visitor = [this, prob](Iterator it_begin,
157  Iterator it_end) {
158  std::vector<Gate*> combination;
159  for (auto it = it_begin; it != it_end; ++it)
160  combination.push_back(it->first);
161 
162  auto ccf_event = std::make_unique<CcfEvent>(std::move(combination), this);
163 
164  for (auto it = it_begin; it != it_end; ++it)
165  it->second.Add(ccf_event.get());
166 
167  ccf_event->expression(prob);
168  ccf_events_.emplace_back(std::move(ccf_event));
169 
170  return false;
171  };
172  ext::for_each_combination(proxy_gates.begin(),
173  std::next(proxy_gates.begin(), level),
174  proxy_gates.end(), combination_visitor);
175  }
176 
177  // Assign formulas to the proxy gates.
178  for (std::pair<Gate*, Formula::ArgSet>& gate : proxy_gates) {
179  assert(gate.second.size() >= 2);
180  gate.first->formula(std::make_unique<Formula>(kOr, std::move(gate.second)));
181  }
182 }

184 CcfGroup::ExpressionMap BetaFactorModel::CalculateProbabilities() {
185  assert(CcfGroup::factors().size() == 1);
186  assert(CcfGroup::members().size() == CcfGroup::factors().front().first);
187 
188  ExpressionMap probabilities;
189 
190  Expression* beta = CcfGroup::factors().begin()->second;
191  probabilities.emplace_back( // (1 - beta) * Q
192  1, CcfGroup::Register<Mul>(
193  {CcfGroup::Register<Sub>({&ConstantExpression::kOne, beta}),
194  CcfGroup::distribution()}));
195 
196  probabilities.emplace_back( // beta * Q
197  CcfGroup::factors().front().first,
198  CcfGroup::Register<Mul>({beta, CcfGroup::distribution()}));
199  return probabilities;
200 }

204 /// Helper function to calculate reciprocal of
205 /// nCk (n-choose-k) combination.
206 ///
207 /// @param[in] n The total number elements.
208 /// @param[in] k Subset size.
209 ///
210 /// @returns 1 / nCk
211 double CalculateCombinationReciprocal(int n, int k) {
212  assert(n >= 0);
213  assert(k >= 0);
214  assert(n >= k);
215  if (n - k > k)
216  k = n - k;
217  double result = 1;
218  for (int i = 1; i <= n - k; ++i) {
219  result *= static_cast<double>(i) / static_cast<double>(k + i);
220  }
221  return result;
222 }

226 CcfGroup::ExpressionMap MglModel::CalculateProbabilities() {
227  ExpressionMap probabilities;
228  int max_level = CcfGroup::factors().back().first;
229  assert(CcfGroup::factors().size() == max_level - 1);
230 
231  int num_members = CcfGroup::members().size();
232  for (int i = 0; i < max_level; ++i) {
233  double mult = CalculateCombinationReciprocal(num_members - 1, i);
234  std::vector<Expression*> args;
235  args.push_back(CcfGroup::Register<ConstantExpression>(mult));
236  for (int j = 0; j < i; ++j) {
237  args.push_back(CcfGroup::factors()[j].second);
238  }
239  if (i < max_level - 1) {
240  args.push_back(CcfGroup::Register<Sub>(
241  {&ConstantExpression::kOne, CcfGroup::factors()[i].second}));
242  }
243  args.push_back(CcfGroup::distribution());
244  probabilities.emplace_back(i + 1, CcfGroup::Register<Mul>(std::move(args)));
245  }
246  assert(probabilities.size() == max_level);
247  return probabilities;
248 }

250 CcfGroup::ExpressionMap AlphaFactorModel::CalculateProbabilities() {
251  ExpressionMap probabilities;
252  int max_level = CcfGroup::factors().back().first;
253  assert(CcfGroup::factors().size() == max_level);
254  std::vector<Expression*> sum_args;
255  for (const std::pair<int, Expression*>& factor : CcfGroup::factors()) {
256  sum_args.emplace_back(CcfGroup::Register<Mul>(
257  {CcfGroup::Register<ConstantExpression>(factor.first), factor.second}));
258  }
259  Expression* sum = CcfGroup::Register<Add>(std::move(sum_args));
260  int num_members = CcfGroup::members().size();
261 
262  for (int i = 0; i < max_level; ++i) {
263  double mult = CalculateCombinationReciprocal(num_members - 1, i);
264  Expression* level = CcfGroup::Register<ConstantExpression>(i + 1);
265  Expression* fraction =
266  CcfGroup::Register<Div>({CcfGroup::factors()[i].second, sum});
267  Expression* prob = CcfGroup::Register<Mul>(
268  {level, CcfGroup::Register<ConstantExpression>(mult), fraction,
269  CcfGroup::distribution()});
270  probabilities.emplace_back(i + 1, prob);
271  }
272  assert(probabilities.size() == max_level);
273  return probabilities;
274 }

276 void PhiFactorModel::DoValidate() const {
277  double sum = 0;
278  double sum_min = 0;
279  double sum_max = 0;
280  for (const std::pair<int, Expression*>& factor : CcfGroup::factors()) {
281  sum += factor.second->value();
282  Interval interval = factor.second->interval();
283  sum_min += interval.lower();
284  sum_max += interval.upper();
285  }
286  if (!ext::is_close(1, sum, 1e-4) || !ext::is_close(1, sum_min, 1e-4) ||
287  !ext::is_close(1, sum_max, 1e-4)) {
288  SCRAM_THROW(ValidityError("The factors for Phi model CCF must sum to 1."));
289  }
290 }
