108 /// Parses the command-line arguments.
109 ///
110 /// @param[in] argc Count of arguments.
111 /// @param[in] argv Values of arguments.
112 /// @param[out] vm Variables map of program options.

114 /// @returns 0 for success.
115 /// @returns 1 for errored state.
116 /// @returns -1 for information only state like help and version.
117 int ParseArguments(int argc, char* argv[], po::variables_map* vm) {
118  const char* usage = "Usage: scram [options] input-files...";
119  po::options_description desc = ConstructOptions();
120  try {
121  po::store(po::parse_command_line(argc, argv, desc), *vm);
122  } catch (std::exception& err) {
123  std::cerr << "Option error: " << err.what() << "\n\n"
124  << usage << "\n\n"
125  << desc << std::endl;
126  return 1;
127  }
128  po::options_description options("All options with positional input files.");
129  options.add(desc).add_options()("input-files",
130  po::value<std::vector<std::string>>(),
131  "MEF input files with analysis constructs");
132  po::positional_options_description p;
133  p.add("input-files", -1); // All input files are implicit.
134  po::store(
135  po::command_line_parser(argc, argv).options(options).positional(p).run(),
136  *vm);
137  po::notify(*vm);
138 
139  auto print_help = [&usage, &desc](std::ostream& out) {
140  out << usage << "\n\n" << desc << std::endl;
141  };
142 
143  // Process command-line arguments.
144  if (vm->count("help")) {
145  print_help(std::cout);
146  return -1;
147  }
148  if (vm->count("version")) {
149  std::cout << "SCRAM " << SCRAM_VERSION << " (" << SCRAM_GIT_REVISION << ")"
150  << "\n\nDependencies:\n"
151  << " Boost " << BOOST_LIB_VERSION << "\n"
152  << " libxml2 " << LIBXML_DOTTED_VERSION << std::endl;
153  return -1;
154  }
155 
156  if (vm->count("verbosity")) {
157  int level = (*vm)["verbosity"].as<int>();
158  if (level < 0 || level > scram::kMaxVerbosity) {
159  std::cerr << "Log verbosity must be between 0 and "
160  << scram::kMaxVerbosity << ".\n\n";
161  print_help(std::cerr);
162  return 1;
163  }
164  }
165 
166  if (!vm->count("input-files") && !vm->count("project")) {
167  std::cerr << "No input or configuration file is given.\n\n";
168  print_help(std::cerr);
169  return 1;
170  }
171  if ((vm->count("bdd") + vm->count("zbdd") + vm->count("mocus")) > 1) {
172  std::cerr << "Mutually exclusive qualitative analysis algorithms.\n"
173  << "(MOCUS/BDD/ZBDD) cannot be applied at the same time.\n\n";
174  print_help(std::cerr);
175  return 1;
176  }
177  if (vm->count("rare-event") && vm->count("mcub")) {
178  std::cerr << "The rare event and MCUB approximations cannot be "
179  << "applied at the same time.\n\n";
180  print_help(std::cerr);
181  return 1;
182  }
183  return 0;
184 }

194 /// Updates analysis settings from command-line arguments.
195 ///
196 /// @param[in] vm Variables map of program options.
197 /// @param[in,out] settings Pre-configured or default settings.
198 ///
199 /// @throws SettingsError The indication of an error in arguments.
200 /// @throws std::exception vm does not contain a required option.
201 /// At least defaults are expected.
202 void ConstructSettings(const po::variables_map& vm,
203  scram::core::Settings* settings) {
204  if (vm.count("bdd")) {
205  settings->algorithm(scram::core::Algorithm::kBdd);
206  } else if (vm.count("zbdd")) {
207  settings->algorithm(scram::core::Algorithm::kZbdd);
208  } else if (vm.count("mocus")) {
209  settings->algorithm(scram::core::Algorithm::kMocus);
210  }
211  settings->prime_implicants(vm.count("prime-implicants"));
212  // Determine if the probability approximation is requested.
213  if (vm.count("rare-event")) {
214  assert(!vm.count("mcub"));
215  settings->approximation(scram::core::Approximation::kRareEvent);
216  } else if (vm.count("mcub")) {
217  settings->approximation(scram::core::Approximation::kMcub);
218  }
219  SET("time-step", double, time_step);
220  settings->safety_integrity_levels(vm.count("sil"));
221 
222  settings->probability_analysis(vm.count("probability"));
223  settings->importance_analysis(vm.count("importance"));
224  settings->uncertainty_analysis(vm.count("uncertainty"));
225  settings->ccf_analysis(vm.count("ccf"));
226  SET("seed", int, seed);
227  SET("limit-order", int, limit_order);
228  SET("cut-off", double, cut_off);
229  SET("mission-time", double, mission_time);
230  SET("num-trials", int, num_trials);
231  SET("num-quantiles", int, num_quantiles);
232  SET("num-bins", int, num_bins);
233 #ifndef NDEBUG
234  settings->preprocessor = vm.count("preprocessor");
235  settings->print = vm.count("print");
236 #endif
237 }

240 /// Main body of command-line entrance to run the program.
241 ///
242 /// @param[in] vm Variables map of program options.
243 ///
244 /// @throws Error Exceptions specific to SCRAM.
245 /// @throws boost::exception Boost errors with the variables map.
246 /// @throws std::exception All other problems.
247 void RunScram(const po::variables_map& vm) {
248  scram::core::Settings settings; // Analysis settings.
249  std::vector<std::string> input_files;
250  // Get configurations if any.
251  // Invalid configurations will throw.
252  if (vm.count("project")) {
253  scram::Project config(vm["project"].as<std::string>());
254  settings = config.settings();
255  input_files = config.input_files();
256  }
257  // Command-line settings overwrite
258  // the settings from the configurations.
259  ConstructSettings(vm, &settings);
260  if (vm.count("input-files")) {
261  auto cmd_input = vm["input-files"].as<std::vector<std::string>>();
262  input_files.insert(input_files.end(), cmd_input.begin(), cmd_input.end());
263  }
264  // Process input files
265  // into valid analysis containers and constructs.
266  // Throws if anything is invalid.
267  std::unique_ptr<scram::mef::Model> model =
268  scram::mef::Initializer(input_files, settings, vm.count("allow-extern"))
269  .model();
270 #ifndef NDEBUG
271  if (vm.count("serialize"))
272  return Serialize(*model, stdout);
273 #endif
274  if (vm.count("validate"))
275  return; // Stop if only validation is requested.
276 
277  // Initiate risk analysis with the given information.
278  scram::core::RiskAnalysis analysis(model.get(), settings);
279  analysis.Analyze();
280 #ifndef NDEBUG
281  if (vm.count("no-report") || vm.count("preprocessor") || vm.count("print"))
282  return;
283 #endif
284  scram::Reporter reporter;
285  bool indent = vm.count("no-indent") ? false : true;
286  if (vm.count("output")) {
287  reporter.Report(analysis, vm["output"].as<std::string>(), indent);
288  } else {
289  reporter.Report(analysis, stdout, indent);
290  }
291 }

293 /// Callback function to redirect XML library error/warning messages to logging.
294 /// Otherwise, the messages are printed to the standard error.
295 ///
296 /// @param[in] msg The printf-style format string.
297 /// @param[in] ... The variadic arguments for the format string.
298 ///
299 /// @pre The library strictly follows validity conditions of printf.
300 extern "C" void LogXmlError(void* /*ctx*/, const char* msg, ...) noexcept {
301  std::va_list args;
302  va_start(args, msg);
303  SCOPE_EXIT([&args] { va_end(args); });
304 
305  std::va_list args_for_nchar; // Only used to determine the string length.
306  va_copy(args_for_nchar, args);
307  int nchar = std::vsnprintf(nullptr, 0, msg, args_for_nchar);
308  va_end(args_for_nchar);
309  if (nchar < 0) {
310  LOG(scram::ERROR) << "String formatting failure: " << std::strerror(errno);
311  return;
312  }
313 
314  std::vector<char> buffer(nchar + /*null terminator*/ 1);
315  std::vsnprintf(buffer.data(), buffer.size(), msg, args);
316  LOG(scram::WARNING) << buffer.data();
317 }

333 /// Command-line SCRAM entrance.
334 ///
335 /// @param[in] argc Argument count.
336 /// @param[in] argv Argument vector.
337 ///
338 /// @returns 0 for success.
339 /// @returns 1 for errored state.
340 int main(int argc, char* argv[]) {
341  LIBXML_TEST_VERSION
342  xmlInitParser();
343  SCOPE_EXIT(&xmlCleanupParser);
344 
345  xmlGenericErrorFunc xml_error_printer = LogXmlError;
346  initGenericErrorDefaultFunc(&xml_error_printer);
347 
348  try {
349  // Parse command-line options.
350  po::variables_map vm;
351  int ret = ParseArguments(argc, argv, &vm);
352  if (ret == 1)
353  return 1;
354 
355  if (vm.count("verbosity")) {
356  scram::Logger::report_level(
357  static_cast<scram::LogLevel>(vm["verbosity"].as<int>()));
358  }
359 
360  if (ret == 0)
361  RunScram(vm);
362  } catch (const scram::LogicError& err) {
363  LOG(scram::ERROR) << "Logic Error:\n" << boost::diagnostic_information(err);
364  return 1;
365  } catch (const scram::IOError& err) {
366  LOG(scram::DEBUG1) << boost::diagnostic_information(err);
367  std::cerr << boost::core::demangled_name(typeid(err)) << "\n\n";
368  PrintErrorInfo<boost::errinfo_file_name>("File", err);
369  PrintErrorInfo<boost::errinfo_file_open_mode>("Open mode", err);
370  if (const int* errnum = boost::get_error_info<boost::errinfo_errno>(err)) {
371  std::cerr << "Error code: " << *errnum << "\n";
372  std::cerr << "Error string: " << std::strerror(*errnum) << "\n";
373  }
374  std::cerr << "\n" << err.what() << std::endl;
375  return 1;
376  } catch (const scram::Error& err) {
377  using namespace scram; // NOLINT
378  LOG(DEBUG1) << boost::diagnostic_information(err);
379  std::cerr << boost::core::demangled_name(typeid(err)) << "\n\n";
380  PrintErrorInfo<errinfo_value>("Value", err);
381  PrintErrorInfo<boost::errinfo_file_name>("File", err);
382  PrintErrorInfo<boost::errinfo_at_line>("Line", err);
383  PrintErrorInfo<mef::errinfo_connective>("MEF Connective", err);
384  PrintErrorInfo<mef::errinfo_reference>("MEF reference", err);
385  PrintErrorInfo<mef::errinfo_base_path>("MEF base path", err);
386  PrintErrorInfo<mef::errinfo_element_id>("MEF Element ID", err);
387  PrintErrorInfo<mef::errinfo_element_type>("MEF Element type", err);
388  PrintErrorInfo<mef::errinfo_container_id>("MEF Container", err);
389  PrintErrorInfo<mef::errinfo_container_type>("MEF Container type", err);
390  PrintErrorInfo<mef::errinfo_attribute>("MEF Attribute", err);
391  PrintErrorInfo<mef::errinfo_cycle>("Cycle", err);
392  PrintErrorInfo<xml::errinfo_element>("XML element", err);
393  PrintErrorInfo<xml::errinfo_attribute>("XML attribute", err);
394  std::cerr << "\n" << err.what() << std::endl;
395  return 1;
396  } catch (const boost::exception& boost_err) {
397  LOG(scram::ERROR) << "Unexpected Boost Exception:\n"
398  << boost::diagnostic_information(boost_err);
399  return 1;
400  } catch (const std::exception& err) {
401  LOG(scram::ERROR) << "Unexpected Exception: "
402  << boost::core::demangled_name(typeid(err)) << ":\n"
403  << err.what();
404  return 1;
405  }
