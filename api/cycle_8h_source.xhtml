113 /// Traverses nodes with connectors to find a cycle.
114 /// Interrupts the detection at first cycle.
115 /// Nodes get marked.
116 ///
117 /// The connector of the node is retrieved via unqualified call to
118 /// GetConnector(node).
119 ///
120 /// @tparam T The type of nodes in the graph.
121 ///
122 /// @param[in,out] node The node to start with.
123 /// @param[out] cycle If a cycle is detected,
124 /// it is given in reverse,
125 /// ending with the cycle node.
126 ///
127 /// @returns True if a cycle is found.
128 ///
129 /// @post All traversed nodes are marked with non-clear marks.
130 template <class T>
131 bool DetectCycle(T* node, std::vector<T*>* cycle) {
132  if (!node->mark()) {
133  node->mark(NodeMark::kTemporary);
134  if (ContinueConnector(GetConnector(node), cycle)) {
135  if (cycle->size() == 1 || cycle->back() != cycle->front())
136  cycle->push_back(node);
137  return true;
138  }
139  node->mark(NodeMark::kPermanent);
140  } else if (node->mark() == NodeMark::kTemporary) {
141  assert(cycle->empty() && "The report container must be provided empty.");
142  cycle->push_back(node);
143  return true;
144  }
145  assert(node->mark() == NodeMark::kPermanent);
146  return false;
147 }

149 template <class T, class N>
150 bool ContinueConnector(T* connector, std::vector<N*>* cycle) {
151  for (N* node : GetNodes(connector)) {
152  if (DetectCycle(node, cycle))
153  return true;
154  }
155  for (auto* link : GetConnectors(connector)) {
156  if (ContinueConnector(link, cycle))
157  return true;
158  }
159  return false;
160 }

186 /// Prints the detected cycle from the output
187 /// produced by cycle detection functions.
188 ///
189 /// @tparam T The node type with GetUniqueName(T*) defined.
190 ///
191 /// @param[in] cycle Cycle containing nodes in reverse order.
192 ///
193 /// @returns String representation of the cycle.
194 template <class T>
195 std::string PrintCycle(const std::vector<T*>& cycle) {
196  assert(cycle.size() > 1);
197  assert(cycle.front() == cycle.back() && "No cycle is provided.");
198  return boost::join(
199  boost::adaptors::reverse(cycle) |
200  boost::adaptors::transformed(
201  [](T* node) -> decltype(auto) { return GetUniqueName(node); }),
202  "->");
203 }

205 /// Checks for cycles in a model constructs.
206 ///
207 /// @tparam T The type of the node.
208 /// @tparam SinglePassRange The range type with nodes.
209 ///
210 /// @param[in] container The range with nodes to be tested.
211 /// @param[in] type The type of nodes for error messages.
212 ///
213 /// @throws CycleError A cycle is detected in the graph of nodes.
214 template <class T, class SinglePassRange>
215 void CheckCycle(const SinglePassRange& container, const char* type) {
216  std::vector<T*> cycle;
217  for (T& node : container) {
218  if (DetectCycle(&node, &cycle)) {
219  SCRAM_THROW(CycleError()) << errinfo_element(GetUniqueName(&node), type)
220  << errinfo_cycle(PrintCycle(cycle));
221  }
222  }
223 }
