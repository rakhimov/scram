104 void Formula::Swap(ArgEvent current, ArgEvent other) {
105  auto it = boost::find_if(args_.data(), [&current](const Arg& arg) {
106  return arg.event == current;
107  });
108  if (it == args_.data().end())
109  SCRAM_THROW(LogicError("The current event is not in the formula."));
110 
111  Event* base = ext::as<Event*>(other);
112  if (ext::any_of(args_.data(), [&current, &base](const Arg& arg) {
113  return arg.event != current &&
114  ext::as<Event*>(arg.event)->id() == base->id();
115  })) {
116  SCRAM_THROW(DuplicateElementError())
117  << errinfo_element(base->id(), "event");
118  }
119 
120  ValidateNesting({it->complement, other});
121 
122  if (!base->usage())
123  base->usage(true);
124 
125  it->event.swap(other);
126 }

128 void Formula::ValidateMinMaxNumber(std::optional<int> min_number,
129  std::optional<int> max_number) {
130  assert(!min_number ||
131  std::numeric_limits<decltype(min_number_)>::max() >= *min_number);
132  assert(!max_number ||
133  std::numeric_limits<decltype(max_number_)>::max() >= *max_number);
134 
135  if (min_number) {
136  if (*min_number < 0)
137  SCRAM_THROW(LogicError("The min number cannot be negative."))
138  << errinfo_value(std::to_string(*min_number));
139 
140  if (connective_ != kAtleast && connective_ != kCardinality) {
141  SCRAM_THROW(
142  LogicError("The min number can only be defined for 'atleast' "
143  "or 'cardinality' connective."))
144  << errinfo_connective(kConnectiveToString[connective_]);
145  }
146  }
147 
148  if (max_number) {
149  if (*max_number < 0)
150  SCRAM_THROW(LogicError("The max number cannot be negative."))
151  << errinfo_value(std::to_string(*max_number));
152 
153  if (connective_ != kCardinality) {
154  SCRAM_THROW(LogicError(
155  "The max number can only be defined for 'cardinality' connective."))
156  << errinfo_connective(kConnectiveToString[connective_]);
157  }
158 
159  if (min_number && *min_number > *max_number)
160  SCRAM_THROW(ValidityError(
161  "The connective min number cannot be greater than max number."))
162  << errinfo_value(std::to_string(*min_number) + " > " +
163  std::to_string(*max_number));
164  }
165 }

167 void Formula::ValidateConnective(std::optional<int> min_number,
168  std::optional<int> max_number) {
169  switch (connective_) {
170  case kAnd:
171  case kOr:
172  case kNand:
173  case kNor:
174  if (args_.size() < 2)
175  SCRAM_THROW(
176  ValidityError("The connective must have 2 or more arguments."));
177  break;
178  case kNot:
179  case kNull:
180  if (args_.size() != 1)
181  SCRAM_THROW(
182  ValidityError("The connective must have only one argument."));
183  break;
184  case kXor:
185  case kIff:
186  case kImply:
187  if (args_.size() != 2)
188  SCRAM_THROW(
189  ValidityError("The connective must have exactly 2 arguments."));
190  break;
191  case kAtleast:
192  if (!min_number)
193  SCRAM_THROW(
194  ValidityError("The connective requires min number for its args."));
195 
196  if (min_number_ < 2)
197  SCRAM_THROW(ValidityError("Min number cannot be less than 2."))
198  << errinfo_value(std::to_string(min_number_));
199 
200  if (args_.size() <= min_number_) {
201  SCRAM_THROW(
202  ValidityError("The connective must have more arguments "
203  "than its min number."))
204  << errinfo_value(std::to_string(args_.size()) +
205  " <= " + std::to_string(min_number_));
206  }
207  break;
208  case kCardinality:
209  if (!min_number || !max_number)
210  SCRAM_THROW(ValidityError(
211  "The connective requires min and max numbers for args."));
212  if (args_.empty())
213  SCRAM_THROW(
214  ValidityError("The connective requires one or more arguments."));
215 
216  if (args_.size() < max_number_)
217  SCRAM_THROW(
218  ValidityError("The connective max number cannot be greater than "
219  "the number of arguments."))
220  << errinfo_value(std::to_string(max_number_) + " > " +
221  std::to_string(args_.size()));
222  }
223 }

225 void Formula::ValidateNesting(const Arg& arg) {
226  if (arg.complement) {
227  if (connective_ == kNull || connective_ == kNot)
228  SCRAM_THROW(LogicError("Invalid nesting of a complement arg."));
229  }
230  if (connective_ == kNot) {
231  if (arg.event == ArgEvent(&HouseEvent::kTrue) ||
232  arg.event == ArgEvent(&HouseEvent::kFalse)) {
233  SCRAM_THROW(LogicError("Invalid nesting of a constant arg."));
234  }
235  }
