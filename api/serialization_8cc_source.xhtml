39 void Serialize(const Model& model, const std::string& file) {
40  std::unique_ptr<std::FILE, decltype(&std::fclose)> fp(
41  std::fopen(file.c_str(), "w"), &std::fclose);
42  try {
43  if (!fp) {
44  SCRAM_THROW(IOError("Cannot open the output file for serialization."))
45  << boost::errinfo_errno(errno) << boost::errinfo_file_open_mode("w");
46  }
47  Serialize(model, fp.get());
48  } catch (IOError& err) {
49  err << boost::errinfo_file_name(file);
50  throw;
51  }
52 }

56 void SerializeLabelAndAttributes(const Element& element,
57  xml::StreamElement* xml_element) {
58  if (element.label().empty() == false)
59  xml_element->AddChild("label").AddText(element.label());
60  if (element.attributes().empty() == false) {
61  xml::StreamElement attributes_container =
62  xml_element->AddChild("attributes");
63  for (const Attribute& attribute : element.attributes()) {
64  xml::StreamElement attribute_element =
65  attributes_container.AddChild("attribute");
66  attribute_element.SetAttribute("name", attribute.name());
67  attribute_element.SetAttribute("value", attribute.value());
68  if (attribute.type().empty() == false)
69  attribute_element.SetAttribute("type", attribute.type());
70  }
71  }
72 }

79 void Serialize(const Formula& formula, xml::StreamElement* parent) {
80  auto stream_event = [](const Formula::ArgEvent& event,
81  xml::StreamElement* xml) {
82  xml->AddChild("event").SetAttribute(
83  "name", ext::as<const mef::Event*>(event)->name());
84  };
85  auto stream_arg = [&stream_event](const Formula::Arg& arg,
86  xml::StreamElement* xml) {
87  if (arg.complement) {
88  xml::StreamElement not_parent = xml->AddChild("not");
89  stream_event(arg.event, &not_parent);
90  } else {
91  stream_event(arg.event, xml);
92  }
93  };
94  if (formula.connective() == kNull) {
95  assert(formula.args().size() == 1);
96  assert(formula.args().front().complement == false);
97  stream_event(formula.args().front().event, parent);
98  } else {
99  xml::StreamElement type_element = [&formula, &parent] {
100  switch (formula.connective()) {
101  case kNot:
102  assert(formula.args().front().complement == false);
103  return parent->AddChild("not");
104  case kAnd:
105  return parent->AddChild("and");
106  case kOr:
107  return parent->AddChild("or");
108  case kNand:
109  return parent->AddChild("nand");
110  case kNor:
111  return parent->AddChild("nor");
112  case kXor:
113  return parent->AddChild("xor");
114  case kAtleast:
115  return [&formula, &parent] {
116  xml::StreamElement atleast = parent->AddChild("atleast");
117  atleast.SetAttribute("min", *formula.min_number());
118  return atleast;
119  }(); // Wrap NRVO into RVO for GCC.
120  default:
121  assert(false && "Unexpected formula");
122  }
123  }();
124  for (const Formula::Arg& arg : formula.args())
125  stream_arg(arg, &type_element);
126  }
127 }

145 void Serialize(const Expression& expression, xml::StreamElement* parent) {
146  if (const auto* constant =
147  dynamic_cast<const ConstantExpression*>(&expression)) {
148  /// @todo Track the original value type of the constant expression.
149  parent->AddChild("float").SetAttribute(
150  "value", const_cast<ConstantExpression*>(constant)->value());
151  } else if (const auto* exponential =
152  dynamic_cast<const Exponential*>(&expression)) {
153  xml::StreamElement xml = parent->AddChild("exponential");
154  assert(exponential->args().size() == 2);
155  for (const Expression* arg : exponential->args())
156  Serialize(*arg, &xml);
157  } else {
158  assert(false && "Unsupported expression");
159  }
160 }

182 void Serialize(const Model& model, std::FILE* out) {
183  xml::Stream xml_stream(out);
184  xml::StreamElement root = xml_stream.root("opsa-mef");
185  if (!model.HasDefaultName())
186  root.SetAttribute("name", model.name());
187  SerializeLabelAndAttributes(model, &root);
188  /// @todo Implement serialization for the following unsupported constructs.
189  assert(model.ccf_groups().empty());
190  assert(model.parameters().empty());
191  assert(model.initiating_events().empty());
192  assert(model.event_trees().empty());
193  assert(model.sequences().empty());
194  assert(model.rules().empty());
195 
196  for (const FaultTree& fault_tree : model.fault_trees())
197  Serialize(fault_tree, &root);
198 
199  xml::StreamElement model_data = root.AddChild("model-data");
200  for (const BasicEvent& basic_event : model.basic_events())
201  Serialize(basic_event, &model_data);
202  for (const HouseEvent& house_event : model.house_events())
203  Serialize(house_event, &model_data);
204 }
