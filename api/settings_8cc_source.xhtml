31 Settings& Settings::algorithm(Algorithm value) noexcept {
32  algorithm_ = value;
33  switch (algorithm_) {
34  case Algorithm::kBdd:
35  approximation(Approximation::kNone);
36  break;
37  default:
38  if (approximation_ == Approximation::kNone)
39  approximation(Approximation::kRareEvent);
40  if (prime_implicants_)
41  prime_implicants(false);
42  }
43  return *this;
44 }
