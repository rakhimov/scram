79 GatePtr Gate::Clone() noexcept {
80  BLOG(DEBUG5, module_) << "WARNING: Cloning module G" << Node::index();
81  assert(!constant() && type_ != kNull);
82  auto clone = std::make_shared<Gate>(type_, &Node::graph()); // The same type.
83  clone->coherent_ = coherent_;
84  clone->min_number_ = min_number_; // Copy min number in case it is K/N.
85  // Getting arguments copied.
86  clone->args_ = args_;
87  clone->gate_args_ = gate_args_;
88  clone->variable_args_ = variable_args_;
89  clone->constant_ = constant_;

101 template <>
102 void Gate::AddConstantArg<true>() noexcept {
103  switch (type_) {
104  case kNull:
105  case kOr:
106  MakeConstant(true);
107  break;
108  case kNand:
109  ReduceLogic(kNot);
110  break;
111  case kAnd:
112  ReduceLogic(kNull);

115  case kNot:

121  break;

123  assert(args_.size() >= 2);

125  --min_number_;

127  type(kOr);
128  break;
129  }

360 void Gate::ProcessAtleastGateDuplicateArg(int index) noexcept {
361  LOG(DEBUG5) << "Handling special case of K/N duplicate argument!";
362  assert(type_ == kAtleast);
363  // This is a very special handling of K/N duplicates.
364  // @(k, [x, x, y_i]) = x & @(k-2, [y_i]) | @(k, [y_i])
365  assert(min_number_ > 1);
366  assert(args_.size() >= min_number_);
367  if (args_.size() == 2) { // @(2, [x, x, z]) = x
368  assert(min_number_ == 2);
369  this->EraseArg(index);
370  this->type(kNull);
371  return;
372  }
373  if (min_number_ == args_.size()) { // @(k, [y_i]) is NULL set.
374  assert(min_number_ > 2 && "Corrupted number of gate arguments.");
375  GatePtr clone_two = this->Clone();
376  clone_two->min_number(min_number_ - 2); // @(k-2, [y_i])
377  this->EraseArgs();
378  this->type(kAnd);
379  clone_two->TransferArg(index, shared_from_this()); // Transferred the x.
380  if (clone_two->min_number() == 1)
381  clone_two->type(kOr);
382  this->AddArg(clone_two);
383  return;
384  }
385  assert(args_.size() > 2);
386  GatePtr clone_one = this->Clone(); // @(k, [y_i])
387 
388  this->EraseArgs(); // The main gate turns into OR with x.
389  type(kOr);
390  this->AddArg(clone_one);
391  if (min_number_ == 2) { // No need for the second K/N gate.
392  clone_one->TransferArg(index, shared_from_this()); // Transferred the x.
393  assert(this->args_.size() == 2);
394  } else {
395  // Create the AND gate to combine with the duplicate node.
396  auto and_gate = std::make_shared<Gate>(kAnd, &Node::graph());
397  this->AddArg(and_gate);
398  clone_one->TransferArg(index, and_gate); // Transferred the x.
399 
400  // Have to create the second K/N for min_number > 2.
401  GatePtr clone_two = clone_one->Clone();
402  clone_two->min_number(min_number_ - 2); // @(k-2, [y_i])
403  if (clone_two->min_number() == 1)
404  clone_two->type(kOr);
405  and_gate->AddArg(clone_two);
406 
407  assert(and_gate->args().size() == 2);
408  assert(this->args_.size() == 2);
409  }
410  assert(clone_one->min_number() <= clone_one->args().size());
411  if (clone_one->args().size() == clone_one->min_number())
412  clone_one->type(kAnd);
413 }

415 void Gate::ProcessComplementArg(int index) noexcept {
416  assert(type_ != kNot && type_ != kNull);
417  assert(args_.count(-index));
418  LOG(DEBUG5) << "Handling complement argument for G" << Node::index();
419  switch (type_) {
420  case kNor:
421  case kAnd:
422  MakeConstant(false);
423  break;
424  case kNand:
425  case kXor:
426  case kOr:
427  MakeConstant(true);
428  break;
429  case kAtleast:
430  LOG(DEBUG5) << "Handling special case of K/N complement argument!";
431  assert(min_number_ > 1 && "Min number is wrong.");
432  assert((args_.size() + 1) > min_number_ && "Malformed K/N gate.");
433  // @(k, [x, x', y_i]) = @(k-1, [y_i])
434  EraseArg(-index);
435  --min_number_;
436  if (args_.size() == 1) {
437  type(kNull);
438  } else if (min_number_ == 1) {
439  type(kOr);
440  } else if (min_number_ == args_.size()) {
441  type(kAnd);
442  }
443  break;
444  default:
445  assert(false && "Unexpected gate type for complement arg processing.");
446  }
447 }

457 Pdag::Pdag(const mef::Gate& root, bool ccf, const mef::Model* model) noexcept
458  : Pdag() {
459  TIMER(DEBUG2, "PDAG Construction");
460  ProcessedNodes nodes;
461  GatherVariables(root.formula(), ccf, &nodes);
462  if (model) { // Process substitution variables.
463  for (const mef::Substitution& substitution : model->substitutions())
464  GatherVariables(substitution, ccf, &nodes);
465  }
466 
467  root_ = ConstructGate(root.formula(), ccf, &nodes);
468 
469  if (model) { // Process substitution application.
470  auto application = std::make_shared<Gate>(kAnd, this);
471  for (const mef::Substitution& substitution : model->substitutions()) {
472  if (substitution.declarative()) { // Apply declarative substitutions.
473  application->AddArg(ConstructSubstitution(substitution, ccf, &nodes));
474  } else { // Gather non-declarative substitutions for analysis.
475  CollectSubstitution(substitution, &nodes);
476  }
477  }
478  if (!application->args().empty()) {
479  application->AddArg(root_);
480  root_ = application;
481  coherent_ = false;
482  }
483  }
484 }

554 template <class T>
555 void Pdag::AddArg(const GatePtr& parent, const T& event, bool complement,
556  bool ccf, ProcessedNodes* nodes) noexcept {
557  if constexpr (std::is_same_v<T, mef::HouseEvent>) {
558  (void)ccf;
559  (void)nodes;
560  // Create unique pass-through gates to hold the construction invariant.
561  auto null_gate = std::make_shared<Gate>(kNull, this);
562  null_gate->AddArg(constant_, complement ^ !event.state());
563  parent->AddArg(null_gate);
564  null_gates_.push_back(null_gate);
565 
566  } else if constexpr (std::is_same_v<T, mef::Gate>) { // NOLINT
567  GatePtr& pdag_gate = nodes->gates.find(&event)->second;
568  if (!pdag_gate) {
569  pdag_gate = ConstructGate(event.formula(), ccf, nodes);
570  }
571  parent->AddArg(pdag_gate, complement);
572 
573  } else {
574  static_assert(std::is_same_v<T, mef::BasicEvent>);
575 
576  if (ccf && event.HasCcf()) { // Replace with a CCF gate.
577  AddArg(parent, event.ccf_gate(), complement, ccf, nodes);
578  } else {
579  VariablePtr& var = nodes->variables.find(&event)->second;
580  assert(var && "Uninitialized variable.");
581  parent->AddArg(var, complement);
582  }
583  }
584 }

597 GatePtr Pdag::ConstructGate(const mef::Formula& formula, bool ccf,
598  ProcessedNodes* nodes) noexcept {
599  static_assert(kNumConnectives == 8, "Unspecified formula connectives.");
600  static_assert(kNumConnectives < mef::kNumConnectives,
601  "MEF and Core connective mismatch.");
602  static_assert(CheckConnectiveEnums(),
603  "mef::Connective must map to core::Connective.");
604 
605  if (formula.connective() >= kNumConnectives)
606  return ConstructComplexGate(formula, ccf, nodes);
607 
608  Connective type = static_cast<Connective>(formula.connective());
609  auto parent = std::make_shared<Gate>(type, this);
610 
611  if (type != kOr && type != kAnd)
612  normal_ = false;
613 
614  switch (type) {
615  case kNot:
616  case kNand:
617  case kNor:
618  case kXor:
619  coherent_ = false;
620  break;
621  case kAtleast:
622  parent->min_number(*formula.min_number());
623  break;
624  case kNull:
625  null_gates_.push_back(parent);
626  break;
627  default:
628  assert((type == kOr || type == kAnd) && "Unexpected gate type.");
629  }
630  for (const mef::Formula::Arg& arg : formula.args()) {
631  if (arg.complement)
632  coherent_ = false;
633  AddArg(parent, arg.event, arg.complement, ccf, nodes);
634  }
635 
636  return parent;
637 }

639 GatePtr Pdag::ConstructComplexGate(const mef::Formula& formula, bool ccf,
640  ProcessedNodes* nodes) noexcept {
641  assert(formula.connective() >= kNumConnectives);
642  coherent_ = false;
643  switch (formula.connective()) {
644  case mef::kIff: {
645  assert(formula.args().size() == 2);
646  normal_ = false;
647  auto parent = std::make_shared<Gate>(kNull, this);
648  auto arg_gate = std::make_shared<Gate>(kXor, this);
649 
650  for (const mef::Formula::Arg& arg : formula.args()) {
651  AddArg(arg_gate, arg.event, arg.complement, ccf, nodes);
652  }
653  parent->AddArg(arg_gate, /*complement=*/true);
654  null_gates_.push_back(parent);
655  return parent;
656  }
657  case mef::kImply: {
658  assert(formula.args().size() == 2);
659  auto parent = std::make_shared<Gate>(kOr, this);
660  AddArg(parent, formula.args().front().event,
661  !formula.args().front().complement, ccf, nodes);
662  AddArg(parent, formula.args().back().event,
663  formula.args().back().complement, ccf, nodes);
664  return parent;
665  }
666  case mef::kCardinality: {
667  assert(formula.args().size() >= *formula.max_number());
668  assert(*formula.min_number() <= *formula.max_number());
669  normal_ = false;
670  auto parent = std::make_shared<Gate>(kAnd, this);
671  auto first_arg = std::make_shared<Gate>(kAtleast, this);
672  first_arg->min_number(*formula.min_number());
673  for (const mef::Formula::Arg& arg : formula.args()) {
674  AddArg(first_arg, arg.event, arg.complement, ccf, nodes);
675  }
676  auto second_arg = first_arg->Clone();
677  second_arg->NegateArgs();
678  second_arg->min_number(formula.args().size() - *formula.max_number());
679 
680  auto well_form = [](Gate* atleast) {
681  if (atleast->min_number() == 0) {
682  atleast->MakeConstant(true);
683  } else if (atleast->min_number() == 1) {
684  atleast->type(kOr);
685  } else if (atleast->min_number() == atleast->args().size()) {
686  atleast->type(kAnd);
687  }
688  };
689  well_form(first_arg.get());
690  well_form(second_arg.get());
691 
692  parent->AddArg(first_arg);
693  parent->AddArg(second_arg);
694  return parent;
695  }
696  default:
697  assert(false && "Unexpected connective for complex gates.");
698  }

716 void Pdag::CollectSubstitution(const mef::Substitution& substitution,
717  ProcessedNodes* nodes) noexcept {
718  assert(!substitution.declarative() && "Only non-declarative substitutions.");
719  int target = [&substitution, &nodes] {
720  if (auto* event = std::get_if<mef::BasicEvent*>(&substitution.target()))
721  return nodes->variables.find(*event)->second->index();
722 
723  assert(*std::get_if<bool>(&substitution.target()) && "Invalid delete term");
724  return 0;
725  }();
726 
727  std::vector<int> source;
728  for (const mef::BasicEvent* event : substitution.source())
729  source.push_back(nodes->variables.find(event)->second->index());
730 
731  switch (substitution.hypothesis().connective()) {
732  case mef::kNull:
733  case mef::kAnd: {
734  std::vector<int> args;
735  for (const mef::Formula::Arg& arg : substitution.hypothesis().args()) {
736  assert(!arg.complement);
737  auto* event = std::get_if<mef::BasicEvent*>(&arg.event);
738  assert(event);
739  args.push_back(nodes->variables.find(*event)->second->index());
740  }
741  substitutions_.push_back({std::move(args), std::move(source), target});
742  break;
743  }
744  case mef::kOr: {
745  for (const mef::Formula::Arg& arg : substitution.hypothesis().args()) {
746  assert(!arg.complement);
747  auto* event = std::get_if<mef::BasicEvent*>(&arg.event);
748  assert(event);
749  substitutions_.push_back(
750  {{nodes->variables.find(*event)->second->index()}, source, target});
751  }
752  break;
753  }
754  default:
755  assert(false && "Only simple formulas for non-declarative substitutions");
756  }

952 /// Provides proper formatting clues for gate formulas.
953 ///
954 /// @param[in] gate The gate with the formula to be printed.
955 ///
956 /// @returns The beginning, operator, and end strings for the formula.
957 FormulaSig GetFormulaSig(const Gate& gate) {
958  FormulaSig sig = {"(", "", ")"}; // Defaults for most gate types.
959 
960  switch (gate.type()) {
961  case kNand:
962  sig.begin = "~("; // Fall-through to AND gate.
963  case kAnd:
964  sig.op = " & ";
965  break;
966  case kNor:
967  sig.begin = "~("; // Fall-through to OR gate.
968  case kOr:
969  sig.op = " | ";
970  break;
971  case kXor:
972  sig.op = " ^ ";
973  break;
974  case kNot:
975  sig.begin = "~("; // Parentheses are for cases of NOT(NOT Arg).
976  break;
977  case kNull:
978  sig.begin = ""; // No need for the parentheses.
979  sig.end = "";
980  break;
981  case kAtleast:
982  sig.begin = "@(" + std::to_string(gate.min_number()) + ", [";
983  sig.op = ", ";
984  sig.end = "])";
985  break;
986  }
987  return sig;
988 }
