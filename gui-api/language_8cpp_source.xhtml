43 std::vector<std::string> translations()
44 {
45  namespace fs = boost::filesystem;
46 
47  std::vector<std::string> result;
48  for (fs::directory_iterator it(translationsPath()), it_end; it != it_end;
49  ++it) {
50  if (!fs::is_regular_file(it->status()))
51  continue;
52 
53  std::string filename = it->path().filename().string();
54  const char suffix[] = ".qm";
55  const char prefix[] = "scramgui_";
56 
57  if (!boost::starts_with(filename, prefix)
58  || !boost::ends_with(filename, suffix))
59  continue;
60 
61  filename.erase(filename.size() - std::strlen(suffix));
62  filename.erase(0, std::strlen(prefix));
63  if (filename != "en")
64  result.push_back(std::move(filename));
65  }
66  return result;
67 }
