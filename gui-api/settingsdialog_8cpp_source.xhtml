40 core::Settings SettingsDialog::settings() const
41 {
42  core::Settings result;
43  try {
44  if (ui->bdd->isChecked()) {
45  result.algorithm(core::Algorithm::kBdd);
46  } else if (ui->zbdd->isChecked()) {
47  result.algorithm(core::Algorithm::kZbdd);
48  } else {
49  GUI_ASSERT(ui->mocus->isChecked(), result);
50  result.algorithm(core::Algorithm::kMocus);
51  }
52 
53  result.prime_implicants(ui->primeImplicants->isChecked());
54  result.probability_analysis(ui->probability->isChecked());
55  result.importance_analysis(ui->importance->isChecked());
56 
57  if (ui->approximationsBox->isChecked() == false) {
58  result.approximation(core::Approximation::kNone);
59  } else if (ui->rareEvent->isChecked()) {
60  result.approximation(core::Approximation::kRareEvent);
61  } else {
62  GUI_ASSERT(ui->mcub->isChecked(), result);
63  result.approximation(core::Approximation::kMcub);
64  }
65 
66  result.limit_order(ui->productOrder->value());
67  result.mission_time(ui->missionTime->value());
68  } catch (const Error &err) {
69  GUI_ASSERT(false && err.what(), result);
70  }
71  return result;
72 }

74 void SettingsDialog::setupState(const core::Settings &initSettings)
75 {
76  ui->primeImplicants->setChecked(initSettings.prime_implicants());
77  ui->probability->setChecked(initSettings.probability_analysis());
78  ui->importance->setChecked(initSettings.importance_analysis());
79  ui->missionTime->setValue(initSettings.mission_time());
80  ui->productOrder->setValue(initSettings.limit_order());
81 
82  switch (initSettings.algorithm()) {
83  case core::Algorithm::kBdd:
84  ui->bdd->setChecked(true);
85  break;
86  case core::Algorithm::kZbdd:
87  ui->zbdd->setChecked(true);
88  break;
89  case core::Algorithm::kMocus:
90  ui->mocus->setChecked(true);
91  break;
92  }
93 
94  ui->approximationsBox->setChecked(true);
95  switch (initSettings.approximation()) {
96  case core::Approximation::kNone:
97  ui->approximationsBox->setChecked(false);
98  break;
99  case core::Approximation::kRareEvent:
100  ui->rareEvent->setChecked(true);
101  break;
102  case core::Approximation::kMcub:
103  ui->mcub->setChecked(true);
104  break;
105  }
106 }

108 void SettingsDialog::setupConnections()
109 {
110  connect(ui->probability, &QAbstractButton::toggled, [this](bool checked) {
111  if (!checked)
112  ui->importance->setChecked(false);
113  });
114  connect(ui->importance, &QAbstractButton::toggled, [this](bool checked) {
115  if (checked)
116  ui->probability->setChecked(true);
117  });
118  connect(ui->bdd, &QAbstractButton::toggled, [this](bool checked) {
119  if (!checked) {
120  ui->approximationsBox->setChecked(true);
121  ui->primeImplicants->setChecked(false);
122  }
123  });
124  connect(ui->primeImplicants, &QAbstractButton::toggled,
125  [this](bool checked) {
126  if (checked) {
127  ui->bdd->setChecked(true);
128  ui->approximationsBox->setChecked(false);
129  }
130  });
131  connect(ui->approximationsBox, &QGroupBox::toggled, [this](bool checked) {
132  if (checked) {
133  ui->primeImplicants->setChecked(false);
134  } else {
135  ui->bdd->setChecked(true);
136  }
137  });
