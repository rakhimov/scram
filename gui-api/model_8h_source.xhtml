331  /// @returns The current connective of the gate.
332  template <typename T = mef::Connective>
333  T type() const
334  {
335  if constexpr (std::is_same_v<T, QString>) {
336  switch (type()) {
337  case mef::kAnd:
338  return _("and");
339  case mef::kOr:
340  return _("or");
341  case mef::kAtleast:
342  //: Also named as 'vote', 'voting or', 'combination', 'combo'.
343  return _("at-least %1").arg(*minNumber());
344  case mef::kXor:
345  return _("xor");
346  case mef::kNot:
347  return _("not");
348  case mef::kNull:
349  //: This is 'pass-through' or 'no-action' gate type.
350  return _("null");
351  case mef::kNand:
352  //: not and.
353  return _("nand");
354  case mef::kNor:
355  //: not or.
356  return _("nor");
357  default:
358  assert(false && "Unsupported connectives.");
359  }
360 
361  } else {

585  ///
586  /// @tparam E The type of the existing Model Event.
587  /// @tparam T The type of the new (target) Event.
588  template <class E, class T>
589  class ChangeEventType : public QUndoCommand
590  {
591  static_assert(!std::is_same_v<E, T>);
592  static_assert(std::is_base_of_v<Element, E>);
593  static_assert(std::is_base_of_v<Element, T>);

595  public:
596  /// Assumes that events have the same ID.
597  ChangeEventType(E *currentEvent,
598  std::unique_ptr<typename T::Origin> newEvent,
599  Model *model, mef::FaultTree *faultTree = nullptr)
600  : QUndoCommand(
601  _("Change the type of event '%1'").arg(currentEvent->id())),
602  m_switchTo{currentEvent, std::make_unique<T>(newEvent.get()),
603  std::move(newEvent)},
604  m_model(model), m_faultTree(faultTree),
605  m_gates(model->parents(currentEvent->data()))
606  {
607  }
608 
609  /// Switches event type to the target one.
610  void redo() override { m_switchFrom = m_switchTo(*this); }
611 
612  /// Switches back the event type to the original one.
613  void undo() override { m_switchTo = m_switchFrom(*this); }
614 
615  private:
616  /// Switches event type from the current one to the next one.
617  template <class Current, class Next>
618  struct Switch
619  {
620  /// @param[in] self The provider of access to the model data.
621  ///
622  /// @returns The reverse operation to switch types to the origin.
623  Switch<Next, Current> operator()(const ChangeEventType &self)
624  {
625  std::unique_ptr<typename Current::Origin> curEvent =
626  self.m_model->m_model->Remove(m_address->data());
627  std::unique_ptr<Current> curProxy =
628  ext::extract(m_address->data(),
629  &self.m_model->template table<Current>());
630  emit self.m_model->removed(m_address);
631  Next *nextAddress = m_proxy.get();
632  self.m_model->m_model->Add(std::move(m_event));
633  self.m_model->template table<Next>().emplace(
634  std::move(m_proxy));
635  emit self.m_model->added(nextAddress);
636  if (self.m_faultTree) {
637  remove(m_address->data(), self.m_faultTree);
638  add(nextAddress->data(), self.m_faultTree);
639  }
640  for (Gate *gate : self.m_gates)
641  gate->data()->formula().Swap(m_address->data(),
642  nextAddress->data());
643 
644  for (Gate *gate : self.m_gates)
645  emit gate->formulaChanged();
646 
647  return {nextAddress, std::move(curProxy), std::move(curEvent)};
648  }
649 
650  Current *m_address; ///< The proxy in the model.
651  std::unique_ptr<Next> m_proxy; ///< The substitute target proxy.
652  std::unique_ptr<typename Next::Origin> m_event; ///< The target.
653  };
