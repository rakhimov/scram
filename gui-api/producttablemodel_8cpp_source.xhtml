30 ProductTableModel::ProductTableModel(const core::ProductContainer &products,
31  bool withProbability, QObject *parent)
32  : QAbstractTableModel(parent), m_withProbability(withProbability)
33 {
34  m_products.reserve(products.size());
35  double sum = 0;
36  for (const core::Product &product : products) {
37  QString members;
38  for (auto it = product.begin(), it_end = product.end(); it != it_end;) {
39  const core::Literal &literal = *it;
40  if (literal.complement)
41  members.append(QStringLiteral("\u00AC"));
42  members.append(QString::fromStdString(literal.event.id()));
43  if (++it != it_end)
44  members.append(QStringLiteral(" \u22C5 "));
45  }
46  double probability = m_withProbability ? product.p() : 0;
47  sum += probability;
48  m_products.push_back(
49  {std::move(members), product.order(), probability});
50  }
51 
52  if (sum == 0)
53  return;
54 
55  for (Product &product : m_products)
56  product.contribution = product.probability / sum;
57 }

90 QVariant ProductTableModel::data(const QModelIndex &index, int role) const
91 {
92  if (!index.isValid())
93  return {};
94  if (role == Qt::TextAlignmentRole && index.column() != 0)
95  return ALIGN_NUMBER_IN_TABLE;
96  if (role != Qt::DisplayRole)
97  return {};
98 
99  GUI_ASSERT(index.row() < m_products.size(), {});
100  const Product &product = m_products[index.row()];
101  switch (index.column()) {
102  case 0:
103  return product.toString;
104  case 1:
105  return product.order;
106  case 2:
107  return product.probability;
108  case 3:
109  return product.contribution;
110  }
111  GUI_ASSERT(false && "unexpected column", {});
112 }
