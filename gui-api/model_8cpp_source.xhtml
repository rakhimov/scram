81 void BasicEvent::SetFlavor::redo()
82 {
83  Flavor cur_flavor = m_basicEvent->flavor();
84  if (m_flavor == cur_flavor)
85  return;
86 
87  mef::Element *mefEvent = m_basicEvent->data();
88  switch (m_flavor) {

90  mefEvent->RemoveAttribute("flavor");

92  case Undeveloped:

95  }

97  emit m_basicEvent->flavorChanged(m_flavor);

149 std::vector<Gate *> Model::parents(mef::Formula::ArgEvent event) const
150 {
151  std::vector<Gate *> result;
152  for (const std::unique_ptr<Gate> &gate : m_gates) {
153  auto it = boost::find_if(gate->args(),
154  [&event](const mef::Formula::Arg &arg) {
155  return arg.event == event;
156  });
157 
158  if (it != gate->args().end())
159  result.push_back(gate.get());
160  }
161  return result;
