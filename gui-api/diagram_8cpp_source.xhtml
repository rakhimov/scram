163 Gate::Gate(model::Gate *event, model::Model *model,
164  std::unordered_map<const mef::Gate *, Gate *> *transfer,
165  QGraphicsItem *parent)
166  : Event(event, parent)
167 {
168  double availableHeight =
169  m_size.height() - m_baseHeight - m_maxSize.height();
170  auto *pathItem = new QGraphicsLineItem(
171  0, 0, 0, (availableHeight - 1) * units().height(), this);
172  pathItem->setPos(0, (m_baseHeight + m_maxSize.height()) * units().height());
173  Event::setTypeGraphics(getGateGraphicsType(event->type()).release());
174  struct
175  {
176  Event *operator()(const mef::BasicEvent *arg)
177  {
178  model::BasicEvent *proxyEvent =
179  m_model->basicEvents().find(arg)->get();
180  switch (proxyEvent->flavor()) {
181  case model::BasicEvent::Basic:
182  return new BasicEvent(proxyEvent, m_parent);
183  case model::BasicEvent::Undeveloped:
184  return new UndevelopedEvent(proxyEvent, m_parent);
185  }
186  GUI_ASSERT(false && "Unexpected event flavor", nullptr);
187  }
188  Event *operator()(const mef::HouseEvent *arg)
189  {
190  return new HouseEvent(m_model->houseEvents().find(arg)->get(),
191  m_parent);
192  }
193  Event *operator()(const mef::Gate *arg)
194  {
195  model::Gate *proxyEvent = m_model->gates().find(arg)->get();
196  if (auto it = ext::find(*m_transfer, arg)) {
197  it->second->addTransferOut();
198  return new TransferIn(proxyEvent, m_parent);
199  }
200  auto *arg_gate =
201  new Gate(proxyEvent, m_model, m_transfer, m_parent);
202  m_transfer->emplace(arg, arg_gate);
203  return arg_gate;
204  }
205 
206  QGraphicsItem *m_parent;
207  decltype(model) m_model;
208  decltype(transfer) m_transfer;
209  } formula_visitor{this, model, transfer};
210  double linkY = (m_size.height() - 1) * units().height();
211  std::vector<std::pair<Event *, QGraphicsLineItem *>> children;
212  for (const mef::Formula::Arg &arg : event->args()) {
213  GUI_ASSERT(!arg.complement, );
214  auto *child = std::visit(formula_visitor, arg.event);
215  auto *link = new QGraphicsLineItem(0, 0, 0, units().height(), this);
216  if (!children.empty())
217  m_width += m_space * units().height();
218  child->moveBy(m_width + child->width() / 2,
219  m_size.height() * units().height());
220  link->moveBy(m_width + child->width() / 2, linkY);
221  m_width += child->width();
222  children.emplace_back(child, link);
223  }
224  // Shift the children left.
225  for (auto &child : children) {
226  child.first->moveBy(-m_width / 2, 0);
227  child.second->moveBy(-m_width / 2, 0);
228  }
229  // Add the planar line to complete the connection.
230  if (children.size() > 1)
231  new QGraphicsLineItem(children.front().first->pos().x(), linkY,
232  children.back().first->pos().x(), linkY, this);
233 }

235 std::unique_ptr<QGraphicsItem> Gate::getGateGraphicsType(mef::Connective type)
236 {
237  static_assert(mef::kNumConnectives > 8, "Unexpected connective changes");
238  switch (type) {
239  case mef::kNull:
240  return std::make_unique<QGraphicsLineItem>(
241  0, 0, 0, m_maxSize.height() * units().height());
242  case mef::kAnd: {
243  double h = m_maxSize.height() * units().height();
244  QPainterPath paintPath;
245  double maxHeight = m_maxSize.height() * units().height();
246  paintPath.moveTo(0, maxHeight);
247  paintPath.arcTo(-h / 2, 0, h, maxHeight * 2, 0, 180);
248  paintPath.closeSubpath();
249  return std::make_unique<QGraphicsPathItem>(paintPath);
250  }
251  case mef::kOr: {
252  QPainterPath paintPath;
253  double x1 = m_maxSize.width() * units().width() / 2;
254  double maxHeight = m_maxSize.height() * units().height();
255  QRectF rectangle(-x1, 0, x1 * 2, maxHeight * 2);
256  paintPath.arcMoveTo(rectangle, 0);
257  paintPath.arcTo(rectangle, 0, 180);
258  double lowerArc = 0.25;
259  rectangle.setHeight(rectangle.height() * lowerArc);
260  rectangle.moveTop(maxHeight * (1 - lowerArc));
261  paintPath.arcMoveTo(rectangle, 0);
262  paintPath.arcTo(rectangle, 0, 180);
263  paintPath.arcMoveTo(rectangle, 90);
264  paintPath.lineTo(0, maxHeight);
265  return std::make_unique<QGraphicsPathItem>(paintPath);
266  }
267  case mef::kAtleast: {
268  double h = m_maxSize.height() * units().height();
269  double a = h / sqrt(3);
270  auto polygon =
271  std::make_unique<QGraphicsPolygonItem>(QPolygonF{{{-a / 2, 0},
272  {a / 2, 0},
273  {a, h / 2},
274  {a / 2, h},
275  {-a / 2, h},
276  {-a, h / 2}}});
277  auto *gate = static_cast<model::Gate *>(m_event);
278  auto *text = new QGraphicsTextItem(QStringLiteral("%1/%2")
279  .arg(gate->minNumber().value())
280  .arg(gate->numArgs()),
281  polygon.get());
282  QFont font = text->font();
283  font.setPointSizeF(1.5 * font.pointSizeF());
284  text->setFont(font);
285  text->setPos(-text->boundingRect().width() / 2,
286  (h - text->boundingRect().height()) / 2);
287  return std::move(polygon);
288  }
289  case mef::kNot: {
290  double h = m_maxSize.height() * units().height();
291  QPainterPath paintPath;
292  paintPath.addEllipse(-units().height() / 2, 0, units().height(),
293  units().height());
294  paintPath.moveTo(0, units().height());
295  double a = h - units().height();
296  paintPath.lineTo(-a / 2, h);
297  paintPath.lineTo(a / 2, h);
298  paintPath.closeSubpath();
299  return std::make_unique<QGraphicsPathItem>(paintPath);
300  }
301  case mef::kXor: {
302  auto orItem = getGateGraphicsType(mef::kOr);
303  double x1 = m_maxSize.width() * units().width() / 2;
304  double h = m_maxSize.height() * units().height();
305  QPainterPath paintPath;
306  paintPath.lineTo(-x1, h);
307  paintPath.moveTo(x1, h);
308  paintPath.lineTo(0, 0);
309  new QGraphicsPathItem(paintPath, orItem.get());
310  return orItem;
311  }
312  case mef::kNor: {
313  auto orItem = getGateGraphicsType(mef::kOr);
314  auto circle = std::make_unique<QGraphicsEllipseItem>(
315  -units().height() / 2, 0, units().height(), units().height());
316  double orHeight = orItem->boundingRect().height();
317  orItem->setScale((orHeight - units().height()) / orHeight);
318  orItem->setPos(0, units().height());
319  orItem.release()->setParentItem(circle.get());
320  return std::move(circle);
321  }
322  case mef::kNand: {
323  auto andItem = getGateGraphicsType(mef::kAnd);
324  auto circle = std::make_unique<QGraphicsEllipseItem>(
325  -units().height() / 2, 0, units().height(), units().height());
326  double andHeight = andItem->boundingRect().height();
327  andItem->setScale((andHeight - units().height()) / andHeight);
328  andItem->setPos(0, units().height());
329  andItem.release()->setParentItem(circle.get());
330  return std::move(circle);
331  }
332  default:
333  GUI_ASSERT(false && "Unexpected gate type", nullptr);
334  }

342 void Gate::addTransferOut()
343 {
