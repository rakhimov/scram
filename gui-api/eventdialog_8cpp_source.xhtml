198 {
199  struct
200  {
201  bool operator()(const mef::Event *) const { return false; }
202  bool operator()(const mef::Gate *arg) const
203  {
204  return m_self->checkCycle(arg);
205  }
206  EventDialog *m_self;
207  } visitor{this};
208 
209  for (const mef::Formula::Arg &arg : gate->formula().args()) {
210  if (ext::as<const mef::Element *>(arg.event) == m_event)
211  return true;

213  return true;
214  }

218 template <class T>
219 mef::FaultTree *EventDialog::getFaultTree(const T *event) const
220 {
221  // Find the fault tree of the first parent gate.
222  auto it = boost::find_if(m_model->gates(), [&event](const mef::Gate &gate) {
223  auto it_arg = boost::find_if(
224  gate.formula().args(), [&event](const mef::Formula::Arg &arg) {
225  return arg.event
226  == mef::Formula::ArgEvent(const_cast<T *>(event));
227  });
228  return it_arg != gate.formula().args().end();
229  });
230  if (it == m_model->gates().end())
231  return nullptr;
232  return getFaultTree(&*it);
233 }

344 void EventDialog::validate()
345 {
346  m_errorBar->clearMessage();
347  emit validated(false);
348 
349  if (nameLine->hasAcceptableInput() == false)
350  return;
351  QString name = nameLine->text();
352  nameLine->setStyleSheet(yellowBackground);
353  try {
354  if (name != m_initName) {
355  m_model->GetEvent(name.toStdString());
356  m_errorBar->showMessage(
357  //: Duplicate event definition in the model.
358  _("The event with name '%1' already exists.").arg(name));
359  return;
360  }
361  } catch (const mef::UndefinedElement &) {
362  }
363 
364  if (!tabFormula->isHidden() && hasFormulaArg(name)) {
365  m_errorBar->showMessage(
366  _("Name '%1' would introduce a self-cycle.").arg(name));
367  return;
368  }
369  nameLine->setStyleSheet({});
370 
371  if (!tabExpression->isHidden() && expressionBox->isChecked()) {
372  switch (stackedWidgetExpressionData->currentIndex()) {
373  case 0:
374  if (constantValue->hasAcceptableInput() == false)
375  return;
376  break;
377  case 1:
378  if (exponentialRate->hasAcceptableInput() == false)
379  return;
380  break;
381  default:
382  GUI_ASSERT(false && "unexpected expression", );
383  }
384  }
385 
386  if (!tabFormula->isHidden()) {
387  int numArgs = argsList->count();
388  switch (static_cast<mef::Connective>(connectiveBox->currentIndex())) {
389  case mef::kNot:
390  case mef::kNull:
391  if (numArgs != 1) {
392  m_errorBar->showMessage(
393  _("%1 connective requires a single argument.")
394  .arg(connectiveBox->currentText()));
395  return;
396  }
397  break;
398  case mef::kAnd:
399  case mef::kOr:
400  case mef::kNand:
401  case mef::kNor:
402  if (numArgs < 2) {
403  m_errorBar->showMessage(
404  _("%1 connective requires 2 or more arguments.")
405  .arg(connectiveBox->currentText()));
406  return;
407  }
408  break;
409  case mef::kXor:
410  if (numArgs != 2) {
411  m_errorBar->showMessage(
412  _("%1 connective requires exactly 2 arguments.")
413  .arg(connectiveBox->currentText()));
414  return;
415  }
416  break;
417  case mef::kAtleast:
418  if (numArgs <= minNumberBox->value()) {
419  int numReqArgs = minNumberBox->value() + 1;
420  m_errorBar->showMessage(
421  //: The number of required arguments is always more than 2.
422  _("%1 connective requires at-least %n arguments.", "",
423  numReqArgs)
424  .arg(connectiveBox->currentText()));
425  return;
426  }
427  break;
428  default:
429  GUI_ASSERT(false && "unexpected connective", );
430  }
431  }
432 
433  if (containerFaultTreeName->isEnabled()) {
434  if (containerFaultTreeName->hasAcceptableInput() == false)
435  return;
436  GUI_ASSERT(typeBox->currentIndex() == ext::one_bit_index(Gate), );
437  QString faultTreeName = containerFaultTreeName->text();
438  if (auto it = ext::find(m_model->fault_trees(),
439  faultTreeName.toStdString())) {
440  GUI_ASSERT(it->top_events().empty() == false, );
441  m_errorBar->showMessage(
442  //: Fault tree redefinition.
443  _("Fault tree '%1' is already defined with a top gate.")
444  .arg(faultTreeName));
445  containerFaultTreeName->setStyleSheet(yellowBackground);
446  return;
447  }
448  }
449  emit validated(true);
450 }
