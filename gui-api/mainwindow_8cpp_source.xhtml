198  if (const std::string *mode =
199  boost::get_error_info<boost::errinfo_file_open_mode>(err)) {

201  }
202  if (const int *errnum = boost::get_error_info<boost::errinfo_errno>(err)) {
203  detail << "Error code: " << *errnum << "\n";
204  detail << "Error string: " << std::strerror(*errnum) << "\n";
205  }

207  message.setDetailedText(QString::fromStdString(detail.str()));

284 bool MainWindow::setProjectFile(const std::string &projectFilePath,

314 bool MainWindow::addInputFiles(const std::vector<std::string> &inputFiles)
315 {
316  static xml::Validator validator(env::install_dir()
317  + "/share/scram/gui.rng");
318 
319  if (inputFiles.empty())
320  return true;
321  if (isWindowModified() && !saveModel())
322  return false;
323 
324  try {
325  std::vector<std::string> allInput = m_inputFiles;
326  allInput.insert(allInput.end(), inputFiles.begin(), inputFiles.end());
327  std::unique_ptr<mef::Model> newModel = [this, &allInput] {
328  return mef::Initializer(allInput, m_settings,
329  /*allow_extern=*/false, &validator)
330  .model();
331  }();
332 
333  for (const mef::FaultTree &faultTree : newModel->fault_trees()) {
334  if (faultTree.top_events().size() != 1) {
335  QMessageBox::critical(
336  this, _("Initialization Error"),
337  //: Single top/root event fault tree are expected by GUI.
338  _("Fault tree '%1' must have a single top-gate.")
339  .arg(QString::fromStdString(faultTree.name())));
340  return false;
341  }
342  }
343 
344  m_model = std::move(newModel);
345  m_inputFiles = std::move(allInput);
346  } catch (const scram::IOError &err) {
347  displayError(err, _("Input file error"), this);
348  return false;
349  } catch (const scram::xml::Error &err) {
350  displayError(err, _("XML Validity Error"), _("Invalid input file"),
351  this);
352  return false;
353  } catch (const scram::mef::ValidityError &err) {
354  displayError(err,
355  //: The error upon initialization from a file.
356  _("Initialization Error"), _("Invalid input model"), this);
357  return false;
358  }
359 
360  emit projectChanged();
361  return true;
362 }
